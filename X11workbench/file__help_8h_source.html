369 static __inline__ int FBWriteFileFromBuffer(const char *szFileName, const char *pBuf, unsigned int cbBuf)
370 {
371 int iRval;
372 file_help_buf_t *pFB;
373 
374 
375  pFB = FBGetFileBufFromBuffer(pBuf, cbBuf);
376 
377  if(!pFB)
378  {
379 // WB_ERROR_PRINT("ERROR - %s - unable to create file buf\n", __FUNCTION__);
380  return -1;
381  }
382 
383  iRval = FBWriteFileBuf(szFileName, pFB);
384 
385  FBDestroyFileBuf(pFB);

388 // {
389 // WB_ERROR_PRINT("ERROR - %s - unable to create file \"%s\" from file buf\n", __FUNCTION__, szFileName);
