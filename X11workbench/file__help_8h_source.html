424 {
425 int iRval;
426 file_help_buf_t *pFB;
427 
428 
429  pFB = FBGetFileBufFromBuffer(pBuf, cbBuf);
430 
431  if(!pFB)
432  {

435  }

438 
439  FBDestroyFileBuf(pFB);
440

442 // {

447 }
