369 // from http://mail.gnome.org/archives/wm-spec-list/2007-March/msg00000.html
370 
371 // The ICCCM provides for focus models for a window:
372 //
373 // Input Model Input Field WM_TAKE_FOCUS
374 // No Input False Absent
375 // Passive True Absent
376 // Locally Active True Present
377 // Globally Active False Present
378 //
379 // No Input - The application never expects to receive focus.
380 //
381 // Passive - The application will get focus but it will not grab it on
382 // its own. Instead it will only get focus if the window manager gives it
383 // to it. The application cannot help determine if it wants the focus or
384 // not at the given time/situation.
385 //
386 // Locally Active - The same as passive, but the application will also
387 // give focus to other windows that it owns of its own free will, without
388 // any interaction with the window manager, using XSetInputFocus.
389 //
390 // Globally Active - The application will not receive focus from the
391 // window manager. Instead it will determine when it wants focus (from
392 // WM_TAKE_FOCUS suggestions given by the window manager, or from events
393 // such as ButtonPress or KeyPress). The application will then acquire
394 // focus on its own using XSetInputFocus.
395 
396  // set title, size hints, and 'WM_HINTS' hints (so WM knows where to put the window and how to set focus)
397  WBSetWMProperties(pNew->wbFW.wID, szTitle, &xsh, &xwmh, NULL);

990 void FWRecalcLayout(Window wID)
991 {
992 WB_RECT rct, rct2;
993 FRAME_WINDOW *pFrameWindow;
994 Window wIDMenu;
995 WBMenuBarWindow *pMB;
996 int i1, iMax;
997 
998 
999  pFrameWindow = (FRAME_WINDOW *)FWGetFrameWindowStruct(wID);
1000 
1001  if(!pFrameWindow)
1002  {
1003  WB_ERROR_PRINT("%s - ERROR - no frame window pointer!\n", __FUNCTION__);
1004  return;
1005  }
1006 
1007  // this callback happens any time I change the window size. Use this
1008  // also to re-size scrollbars and status bars and menus.
1009 
1010  // calculate the actual size of the client area and inform all of the 'Child Frame'
1011  // windows so that they can assign the size and fix scrollbars, etc. as appropriate
1012 
1013  WBGetClientRect(wID, &rct); // get the rectangle for the client area
1014 
1015  // obtain the height and position of the menu (if any), toolbar (if any), and tab area (if any)
1016  // and subtract that from rct.top - then record this as my "client area" in the FRAME_WINDOW
1017  // structure (iClientX, iClientY, iClientWidth, iClientHeight).
1018 
1019  wIDMenu = WBGetMenuWindow(wID);
1020 
1021  if(wIDMenu != None)
1022  {
1023  pMB = MBGetMenuBarWindowStruct(wIDMenu);
1024 
1025  if(pMB)
1026  {
1027  WBGetWindowRect(wIDMenu, &rct2);
1028 
1029  // what I want is the height.
1030  rct.top += rct2.bottom - rct2.top + 1; // regardless of rect, this will be the height, plus 1 pixel
1031  }
1032  }
1033 
1034 
1035  if(!(WBFrameWindow_NO_TABS & pFrameWindow->wbFW.iFlags)) // window has tabs
1036  {
1037  rct.top += TAB_BAR_HEIGHT; // TODO: base this on font height. use the same font as the menu.
1038  }
1039 
1040  // if this window has a status bar on the bottom, subtract an appropriate height
1041  // from the bottom of the client area.
1042 
1043  if(WBFrameWindow_STATUS_BAR & pFrameWindow->wbFW.iFlags) // window has a status bar
1044  {
1045  rct.bottom -= STATUS_BAR_HEIGHT; // TODO: base this on font height. use the same font as the menu.
1046  }
1047 
1048 
1049  // Now that I've calculated the new client rectangle area, assign appropriate thingies
1050  // (but if it matches, bail here, no need for anything else, since nothing changed)
1051 
1052  rct.right -= rct.left; // now it's a width
1053  rct.bottom -= rct.top; // now it's a height
1054 
1055  if(pFrameWindow->wbFW.iClientX == rct.left &&
1056  pFrameWindow->wbFW.iClientY == rct.top &&
1057  pFrameWindow->wbFW.iClientWidth == rct.right &&
1058  pFrameWindow->wbFW.iClientHeight == rct.bottom)
1059  {
1060  return; // nothing changed
1061  }
1062 
1063  // assign the new values. This will be the "true" client area
1064  // of the frame window, excluding "things"
1065  pFrameWindow->wbFW.iClientX = rct.left;
1066  pFrameWindow->wbFW.iClientY = rct.top;
1067  pFrameWindow->wbFW.iClientWidth = rct.right;
1068  pFrameWindow->wbFW.iClientHeight = rct.bottom;
1069 
1070 
1071  // next, inform all of the child frames that I'm different now
1072 
1073  iMax = FWGetNumContWindows(&(pFrameWindow->wbFW));
1074 
1075  for(i1=0; i1 < iMax; i1++)
1076  {
1077  WBChildFrame *pCW = FWGetContainedWindowByIndex(&(pFrameWindow->wbFW), i1);
1078 
1079  if(pCW)
1080  {
1081  FWChildFrameRecalcLayout(pCW); // inform the child frame of the layout change
1082  }
1083  }
1084 
1085  // temporarily use 'error level' so I can always see this
1086  WB_DEBUG_PRINT(DebugLevel_Heavy | DebugSubSystem_Event | DebugSubSystem_Frame,
1087  "%s - %d, %d, %d, %d\n", __FUNCTION__,
1088  pFrameWindow->wbFW.iClientX, pFrameWindow->wbFW.iClientY,
1089  pFrameWindow->wbFW.iClientWidth, pFrameWindow->wbFW.iClientHeight);
1090 
1091  WBInvalidateGeom(wID, NULL, 1); // and, finally, force a re-paint (mostly hidden by child frame, except tabs, borders, etc.)
1092 }
