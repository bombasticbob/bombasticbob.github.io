281 // The ICCCM provides for focus models for a window:
282 //
283 // Input Model Input Field WM_TAKE_FOCUS
284 // No Input False Absent
285 // Passive True Absent
286 // Locally Active True Present
287 // Globally Active False Present
288 //
289 // No Input - The application never expects to receive focus.
290 //
291 // Passive - The application will get focus but it will not grab it on
292 // its own. Instead it will only get focus if the window manager gives it
293 // to it. The application cannot help determine if it wants the focus or
294 // not at the given time/situation.
295 //
296 // Locally Active - The same as passive, but the application will also
297 // give focus to other windows that it owns of its own free will, without
298 // any interaction with the window manager, using XSetInputFocus.
299 //
300 // Globally Active - The application will not receive focus from the
301 // window manager. Instead it will determine when it wants focus (from
302 // WM_TAKE_FOCUS suggestions given by the window manager, or from events
303 // such as ButtonPress or KeyPress). The application will then acquire
304 // focus on its own using XSetInputFocus.
305 
306  // set title, size hints, and 'WM_HINTS' hints (so WM knows where to put the window and how to set focus)
307  WBSetWMProperties(pNew->wbFW.wID, szTitle, &xsh, &xwmh, NULL);
308 
309 #if 0
310  aProto[0] = aWM_DELETE_WINDOW; // for now, that's the only one

377 {

457 int FWDefaultCallback(Window wID, XEvent *pEvent)
458 {
459  FRAME_WINDOW *pFrameWindow = (FRAME_WINDOW *)FWGetFrameWindowStruct(wID);
460  Window wIDMenu;
461  int iRval = 0;
462 #ifndef NO_DEBUG
463  char tbuf[32]; // for keyboard input
464  int nChar = sizeof(tbuf);
465 #endif // NO_DEBUG
466 
467 
468  if(!pFrameWindow)
469  {
470  WB_ERROR_PRINT("%s - ERROR - no frame window pointer!\n", __FUNCTION__);
471  return 0;
472  }
473 
474  wIDMenu = WBGetMenuWindow(wID);
475 
476  // TODO: message re-direction to children BEFORE 'pFWCallback'
477 
478  if(wIDMenu)
479  {
480  switch(pEvent->type)
481  {
482  case ButtonPress:
483  case ButtonRelease:
484  WB_DEBUG_PRINT(DebugLevel_Heavy | DebugSubSystem_Event | DebugSubSystem_Frame,
485  "%s - BUTTON PRESS\n", __FUNCTION__);
486 
487  if(WBPointInWindow(pEvent->xbutton.window, pEvent->xbutton.x, pEvent->xbutton.y, wIDMenu))
488  {
489  return(WBWindowDispatch(wIDMenu, pEvent)); // menu window should handle mousie events
490  }
491 
492  break;
493 
494  case MotionNotify:
495  WB_DEBUG_PRINT(DebugLevel_Heavy | DebugSubSystem_Event | DebugSubSystem_Frame,
496  "%s - MOTION NOTIFY\n", __FUNCTION__);
497 
498  if(WBPointInWindow(pEvent->xmotion.window, pEvent->xmotion.x, pEvent->xmotion.y, wIDMenu))
499  {
500  return(WBWindowDispatch(wIDMenu, pEvent)); // menu window should handle mousie events
501  }
502 
503  break;
504 
505  case ClientMessage: // menus, etc.
506  WB_DEBUG_PRINT(DebugLevel_Heavy | DebugSubSystem_Event | DebugSubSystem_Frame,
507  "%s - CLIENT MESSAGE\n", __FUNCTION__);
508 
509  if(pEvent->xclient.message_type == aMENU_COMMAND)
510  {
511  // check the container window that has the current focus, and THEN
512  // check the frame window for an appropriate handler
513 
514  Window wFocus = FWGetFocusWindow(&(pFrameWindow->wbFW));
515  if(wFocus != None)
516  {
517  int iRet = WBWindowDispatch(wFocus, pEvent);
518  if(iRet) // non-zero return
519  {
520  return iRet; // the handler MUST return non-zero if the message should NOT be processed by the frame!
521  }
522  }
523 
524  if(pFrameWindow->pMenuHandler)
525  {
526  // search for the matching menu or ID - anything above 0x10000 is assumed to be a pointer
527  const WBFWMenuHandler *pHandler = pFrameWindow->pMenuHandler;
528 
529  while(pHandler->lMenuID) // zero marks the end
530  {
531  long lID = pHandler->lMenuID;
532 
533  if(pHandler->lMenuID >= 0x10000L)
534  {
535  lID = XInternAtom(WBGetDefaultDisplay(), (const char *)lID, False);
536 
537  if(!lID)
538  {
539  continue;
540  }
541  }
542 
543  if(pEvent->xclient.data.l[0] == lID)
544  {
545  if(pHandler->callback)
546  {
547  if(pHandler->callback(&(pEvent->xclient)))
548  {
549  return 1;
550  }
551  }
552 
553  return 0; // NOT handled or handler returned zero
554  }
555 
556  pHandler++;
557  }
558  }
559  }
560  else if(pEvent->xclient.message_type == aMENU_UI_COMMAND)
561  {
562  // check 'contained' window for an appropriate UI handler before passing
563  // it off to the frame window's handler
564 
565  Window wFocus = FWGetFocusWindow(&(pFrameWindow->wbFW));
566  if(wFocus != None)
567  {
568  int iRet = WBWindowDispatch(wFocus, pEvent);
569  if(iRet >= 0)
570  {
571  return iRet;
572  }
573 
574  // TODO: determine if a '-1' value really SHOULD grey out the menu choice anyway or
575  // if I should return some OTHER value to differentiate 'grey' from 'no handler'
576  }
577 
578  if(pFrameWindow->pMenuHandler)
579  {
580  const WBFWMenuHandler *pHandler = pFrameWindow->pMenuHandler;
581 
582  while(pHandler->lMenuID) // zero marks the end
583  {
584  long lID = pHandler->lMenuID;
585 
586  if(pHandler->lMenuID >= 0x10000L)
587  {
588  lID = XInternAtom(WBGetDefaultDisplay(), (const char *)lID, False);
589 
590  if(!lID)
591  {
592  continue;
593  }
594  }
595 
596  if(pEvent->xclient.data.l[0] == lID) // a message handler exists
597  {
598  if(pHandler->UIcallback)
599  {
600  WBMenu *pMenu;
601  WBMenuItem *pItem;
602 
603 #if !defined(__SIZEOF_POINTER__) // TODO find a better way
604 #define __SIZEOF_POINTER_ 0
605 #endif
606 #if __SIZEOF_POINTER__ == 4 /* to avoid warnings in 32-bit linux */
607  pMenu = (WBMenu *)pEvent->xclient.data.l[1];
608  pItem = (WBMenuItem *)pEvent->xclient.data.l[3];
609 #else
610  pMenu = (WBMenu *)((unsigned long long)pEvent->xclient.data.l[1] | ((unsigned long long)pEvent->xclient.data.l[2] << 32));
611  pItem = (WBMenuItem *)((unsigned long long)pEvent->xclient.data.l[3] | ((unsigned long long)pEvent->xclient.data.l[4] << 32));
612 #endif
613  // TODO: validate pointers
614 
615  return pHandler->UIcallback(pMenu, pItem);
616  }
617 
618  return 0; // NO UI handler so return '0' [aka 'normal']
619  }
620 
621  pHandler++;
622  }
623  }
624 
625  return -1; // if there's no handler and no UI handler, always return 'disabled'
626  }
627  else
628  {
629  WB_DEBUG_PRINT(DebugLevel_Heavy | DebugSubSystem_Event | DebugSubSystem_Frame,
630  "%s - Client message not handled by frame window\n", __FUNCTION__);
631  }
632 
633  break;
634 
635  default:
636  break;
637  }
638  }
639 
640 
641  if(pFrameWindow->pFWCallback)
642  {
643  // for most messages, if I handle it here, I don't do further processing
644 
645  iRval = (pFrameWindow->pFWCallback)(wID, pEvent);
646 
647  WB_DEBUG_PRINT(DebugLevel_Chatty | DebugSubSystem_Event | DebugSubSystem_Frame,
648  "%s - %s event and user callback returns %d\n", __FUNCTION__, WBEventName(pEvent->type), iRval);
649 
650  if(iRval)
651  {
652  // check message types that I do *NOT* want to 'bail out' for, as well as those
653  // that I _MUST_ bail out for.
654 
655  switch(pEvent->type)
656  {
657  case DestroyNotify: // must process after user callback

659  "%s DestroyNotify and user callback returned a non-zero value\n", __FUNCTION__);

662  // it here also (at the very end)
663 
664  break;
665 
666  case Expose: // no further processing needed, special debug notification

692  case KeyPress:

701  ((XKeyEvent *)pEvent)->state, ((XKeyEvent *)pEvent)->state);
702 #endif // NO_DEBUG
703 
704  // check for menu and hotkey activation.
705 // if(nChar > 0) // only for "real" characters (i.e. not just the ALT key)
706  {
707  WBMenuBarWindow *pMenuBar = MBGetMenuBarWindowStruct(WBGetMenuWindow(wID));
708 
709  if(pMenuBar) // menu bar exists?
710  {

712  "%s call to MBMenuProcessHotKey for menu window %d (%08xH)\n",
713  __FUNCTION__, (int)pMenuBar->wSelf, (int)pMenuBar->wSelf);
714 
715  iRval = MBMenuProcessHotKey(pMenuBar->pMenu, (XKeyEvent *)pEvent);
716  }
717  }
718  }
719  break;
720 
721  case KeyRelease:
722  {
723  // KeyRelease
724 #ifndef NO_DEBUG
725  int iACS = 0, iKey = WBKeyEventProcessKey((XKeyEvent *)pEvent, tbuf, &nChar, &iACS);

732  ((XKeyEvent *)pEvent)->state, ((XKeyEvent *)pEvent)->state);
733

736 
737  }
738 
739  break;

742 
743  case SelectionRequest:
744  case SelectionClear:
745  case SelectionNotify:
746 
747  return FWDoSelectionEvents(&(pFrameWindow->wbFW), wID, wIDMenu, pEvent);
748  }
749  }

758  //

760  //
761  // DESTROY DESTROY DESTROY DESTROY DESTROY
762  // ----------------------------------------

764

766  pEvent->xdestroywindow.window == wID)

844 int FWDoSelectionEvents(WBFrameWindow *pFrameWindow, Window wID, Window wIDMenu, XEvent *pEvent)
845 {
846 Display *pDisplay = WBGetWindowDisplay(wID);
847 FRAME_WINDOW *pFW = (FRAME_WINDOW *)pFrameWindow;
848 
849  // this is what the libXt function 'RequestSelectionValue' does
850  // XConvertSelection(info->ctx->dpy, selection, target, // 'selection' and 'target' are Atoms
851  // info->property, XtWindow(info->widget), info->time);
852 
853 
854  // TEMPORARY: doing what WBDefault does
855 
856  if(pEvent->type == SelectionRequest)
857  {
858  char *p1 = pEvent->xselectionrequest.selection != None ? XGetAtomName(pDisplay, pEvent->xselectionrequest.selection) : NULL;
859  char *p2 = pEvent->xselectionrequest.target != None ? XGetAtomName(pDisplay, pEvent->xselectionrequest.target) : NULL;
860  char *p3 = pEvent->xselectionrequest.property != None ? XGetAtomName(pDisplay, pEvent->xselectionrequest.property) : NULL;
861 
862  WB_ERROR_PRINT("TEMPORARY - %s - SelectionRequest owner=%d requestor=%d selection=%s target=%s property=%s\n",
863  __FUNCTION__,
864  (int)pEvent->xselectionrequest.owner,
865  (int)pEvent->xselectionrequest.requestor,
866  p1 ? p1 : "NULL",
867  p2 ? p2 : "NULL",
868  p3 ? p3 : "NULL");
869 
870  if(p1)
871  {
872  XFree(p1);
873  }
874  if(p2)
875  {
876  XFree(p2);
877  }
878  if(p3)
879  {
880  XFree(p3);
881  }
882 
883  // the default rejects the request. Send a 'SelectionNotify' to indicate the failure
884 
885  if(pEvent->xselectionrequest.owner == wID) // only if I'm the selection owner
886  {
887  XSelectionEvent evtN;
888 
889  memset(&evtN, 0, sizeof(evtN));
890 
891  evtN.type = SelectionNotify;
892  evtN.send_event = True; // 'cause I'm going to use 'XSendEvent' to reply immediately
893  evtN.requestor = pEvent->xselectionrequest.requestor;
894  evtN.selection = pEvent->xselectionrequest.selection;
895  evtN.target = pEvent->xselectionrequest.target;
896  evtN.property = None; // to indicate failure
897  evtN.time = pEvent->xselectionrequest.time; // same time as request (for now)
898 
899  BEGIN_XCALL_DEBUG_WRAPPER
900  XSendEvent(pDisplay, evtN.requestor, False, 0, (XEvent *)&evtN);
901  END_XCALL_DEBUG_WRAPPER
902  }
903 
904  return 1; // "handled"
905  }
906  else if(pEvent->type == SelectionClear)
907  {
908  char *p1 = XGetAtomName(pDisplay, pEvent->xselectionclear.selection);
909 
910  WB_ERROR_PRINT("TEMPORARY - %s - SelectionClear window=%d selection=%s\n",
911  __FUNCTION__,
912  (int)pEvent->xselectionclear.window,
913  p1 ? p1 : "NULL");
914 
915  if(p1)
916  {
917  XFree(p1);
918  }

920  // the default rejects the request. Send a 'SelectionNotify' to indicate the failure
921 
922  if(pEvent->xselectionrequest.owner == wID) // only if I'm the selection owner
923  {
924  XSelectionEvent evtN;
925

953  p2 ? p2 : "NULL",

959  }

962  XFree(p2);
963  }
964  if(p3)
