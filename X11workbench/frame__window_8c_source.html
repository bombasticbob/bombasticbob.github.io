371  XAllocColor(WBGetDefaultDisplay(), colormap, &clrBD);

376  // ---------------------------------------------------------------------------------------
377  // 3D border colors - determine a decent set of border colors for clrBD2 and clrBD3 using
378  // the background color. highlight luminocity will average between black and background
379  // for the shaded color (clrBD3) and between white and background for highlight (clrBD2).
380  // ---------------------------------------------------------------------------------------
381

395

397  PXM_RGBToPixel(PXM_StandardColormapFromColormap(NULL,colormap),
398  &clrBD2); // re-assign pixel element from RGB values
399 
400  // the shaded color should be 3/4 of the way between black and the background color, using the same U and V
401 
402  PXM_YUVToRGB((3 * iY) / 4, iU, iV, &iR, &iG, &iB);
403 
404  RGB255_TO_XCOLOR(iR, iG, iB, clrBD3); // assign new RGB values to the XColor struct
405  PXM_RGBToPixel(PXM_StandardColormapFromColormap(NULL,colormap),
406  &clrBD3); // re-assign pixel element from RGB values
407 
408  XAllocColor(WBGetDefaultDisplay(), colormap, &clrBD2);
409  XAllocColor(WBGetDefaultDisplay(), colormap, &clrBD3);
410 
411  iInitColorFlag = 1;
412  }
