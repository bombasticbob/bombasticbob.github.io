341 
342  if(!pRval)
343  {

388  pRval->iX = /*xsh.x + */2;
389  pRval->iY = /*xsh.y + */2;
390  pRval->iWidth = xsh.width - 4;
391  pRval->iHeight = xsh.height - 4;

435 
436  return(pRval);

482 static void MBMenuPopupHandleMenuItem(Display *pDisplay, Window wID, WBMenuPopupWindow *pSelf, WBMenu *pMenu, WBMenuItem *pItem)
483 {
484  if(pItem->iAction & WBMENU_POPUP_HIGH_BIT)
485  {
486  XClientMessageEvent evt;
487 
488  // post a high-priority message to myself to display the menu
489 
490  bzero(&evt, sizeof(evt));
491  evt.type = ClientMessage;
492  evt.display = pDisplay;
493  evt.window = wID;
494  evt.message_type = aMENU_DISPLAY_POPUP;
495  evt.format = 32;
496  evt.data.l[0] = pItem->iAction & WBMENU_POPUP_MASK;
497  evt.data.l[1] = pItem->iPosition;
498 
499  WBPostPriorityEvent(wID, (XEvent *)&evt);
500  }
501  else
502  {
503  XClientMessageEvent evt;
504 
505  bzero(&evt, sizeof(evt));
506  evt.type = ClientMessage;
507 
508  evt.display = pDisplay;
509  evt.window = pSelf->wOwner;
510  evt.message_type = aMENU_COMMAND;
511  evt.format = 32; // always
512  evt.data.l[0] = pItem->iAction; // menu command message ID
513  evt.data.l[1] = (long)pMenu; // pointer to menu object
514  evt.data.l[2] = wID; // window ID of menu bar
515 
516  WBPostEvent(pSelf->wOwner, (XEvent *)&evt);
517 
518  WB_DEBUG_PRINT(DebugLevel_Heavy | DebugSubSystem_Menu,
519  "%s - Post Event: %08xH %08xH %pH %08xH\n",
520  __FUNCTION__, (int)aMENU_COMMAND, (int)pItem->iAction,
521  pMenu, (int)wID);
522 
523  WBEndModal(wID, pItem->iAction);
524  }
525 }

531  bzero(&evt, sizeof(evt));
