293

335 
336  if(!pRval)
337  return NULL;
338 
339  pRval->ulTag = MENU_POPUP_WINDOW_TAG;
340 
341  pRval->pMenu = pMenu;
342 
343  if(!pRval->pMenu)

347 
348  // this window will be owned by the root, so it will require mouse and keyboard grabs

350

377

379  pRval->iX = /*xsh.x + */2;
380  pRval->iY = /*xsh.y + */2;
381  pRval->iWidth = xsh.width - 4;
382  pRval->iHeight = xsh.height - 4;
383 
384 // WBRegisterWindowCallback(pRval->wSelf, MBMenuPopupEvent); // this must happen first

386 
387  // establish this window as NEVER getting the input focus
388  bzero(&xwmh, sizeof(xwmh));
389  xwmh.flags = InputHint;
390  xwmh.input = 0; // this represents 'None' (or 'Globally Active' if WM_TAKE_FOCUS is present)
391 
392  WBSetWMProperties(pRval->wSelf, NULL, NULL, &xwmh, NULL);
393 
394  WBSetWindowDefaultCursor(pRval->wSelf, XC_hand2); //XC_top_left_arrow);

396 
397  WBCreateWindowDefaultGC(pRval->wSelf, fg, bg);

413  XChangeProperty(pDisplay, pRval->wSelf, a1, XA_WINDOW, 32, PropModeReplace, (unsigned char *)&wIDOwner, 1);//&wIDParent, 1);

422

424 
425  // TODO: set up font, GS, callback, etc.
426 
427  return(pRval);
428 }

463 static void MBMenuPopupHandleMenuItem(Display *pDisplay, Window wID, WBMenuPopupWindow *pSelf, WBMenu *pMenu, WBMenuItem *pItem)
464 {
465  if(pItem->iAction & WBMENU_POPUP_HIGH_BIT)
466  {
467  XClientMessageEvent evt;
468 
469  // post a high-priority message to myself to display the menu
470 
471  bzero(&evt, sizeof(evt));
472  evt.type = ClientMessage;
473  evt.display = pDisplay;
474  evt.window = wID;
475  evt.message_type = aMENU_DISPLAY_POPUP;
476  evt.format = 32;
477  evt.data.l[0] = pItem->iAction & WBMENU_POPUP_MASK;
478  evt.data.l[1] = pItem->iPosition;
479 
480  WBPostPriorityEvent(wID, (XEvent *)&evt);
481  }
482  else
483  {
484  XClientMessageEvent evt;
485 
486  bzero(&evt, sizeof(evt));
487  evt.type = ClientMessage;
488 
489  evt.display = pDisplay;
490  evt.window = pSelf->wOwner;
491  evt.message_type = aMENU_COMMAND;
492  evt.format = 32; // always
493  evt.data.l[0] = pItem->iAction; // menu command message ID
494  evt.data.l[1] = (long)pMenu; // pointer to menu object
495  evt.data.l[2] = wID; // window ID of menu bar
496 
497  WBPostEvent(pSelf->wOwner, (XEvent *)&evt);
498 
499  WB_DEBUG_PRINT(DebugLevel_Heavy | DebugSubSystem_Menu,
500  "%s - Post Event: %08xH %08xH %pH %08xH\n",
501  __FUNCTION__, (int)aMENU_COMMAND, (int)pItem->iAction,
502  pMenu, (int)wID);
503 
504  WBEndModal(wID, pItem->iAction);
505  }
506 }
