73 typedef struct
74 {
75  // the following data members are offsets from 'data' for each component of the menu item
76  // '-1' generically indicates "none"
77  int iMenuItemText; // offset in 'data' to null-byte terminated strings (-1 if none)
78  int iUnderscore; // offset of (first) 'underscore' within menu text (-1 if none)
79  int iTooltipText; // offset in 'data' to null-byte terminated strings (-1 if none)
80  int iHotKey; // hotkey description (-1 if none)
81 
82  // the following data members are numeric properties (not offsets)
83  int iAction; // high bit set for popup; -1 for separator; otherwise, it's a message
84  // if hight bit set, corresponding lower bits are the popup menu ID
85  int nHotKey; // hotkey character translation (0 if none)
86  int iTextWidth; // width of menu text (in pixels; assign '-1' to calculate it)
87  int iPosition; // horizontal/vertical position of menu (in pixels; assign '-1' to calculate it)
88 
89  int nDataSize; // total size of data
90  char data[4]; // data follows
91 
92 } WBMenuItem;
