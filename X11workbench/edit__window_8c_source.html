196 static void InternalCheckEWColorsAndAtoms(void)
197 {
198  Colormap colormap;
199 
200  // *Frame.background, *Frame.foreground, *WmFrame.background, *WmFrame.foreground,
201  // *Form.background, *Form.foreground, *background, *foreground
202 
203  if(aEW_EDIT_CHANGE == None)
204  {
205  aEW_EDIT_CHANGE = WBGetAtom(WBGetDefaultDisplay(), "EW_EDIT_CHANGE");
206  }
207 
208  if(aEW_HOVER_NOTIFY == None)
209  {
210  aEW_HOVER_NOTIFY = WBGetAtom(WBGetDefaultDisplay(), "EW_HOVER_NOTIFY");
211  }
212 
213  if(!iInitColorFlag)
214  {
215  char szFG[16], szBG[16], szAFG[16], szABG[16]; // note colors can typically be up to 13 characters + 0 byte
216 
217  colormap = DefaultColormap(WBGetDefaultDisplay(), DefaultScreen(WBGetDefaultDisplay()));
218 
219  // (these color names and standards have changed *WAY* too many times...)
220 
221  LOAD_COLOR0("*Text.foreground",szFG) else LOAD_COLOR0("*Edit.foreground", szFG)
222  else LOAD_COLOR("*foreground", szFG, "#000000");
223  LOAD_COLOR0("*Text.background",szBG) else LOAD_COLOR0("*Edit.background", szBG)
224  else LOAD_COLOR("*background", szBG, "white"); // pure white background by default
225 
226  LOAD_COLOR("selected_bg_color", szABG, "#0040FF"); // a slightly greenish blue for the 'selected' BG color
227  LOAD_COLOR("selected_fg_color", szAFG, "white"); // white FG when selected
228 
229  WB_ERROR_PRINT("TEMPORARY: %s - edit window colors: FG=%s BG=%s AFG=%s ABG=%s\n", __FUNCTION__,
230  szFG, szBG, szAFG, szABG);
231 
232  XParseColor(WBGetDefaultDisplay(), colormap, szFG, &clrFG);
233  XAllocColor(WBGetDefaultDisplay(), colormap, &clrFG);
234  XParseColor(WBGetDefaultDisplay(), colormap, szBG, &clrBG);
235  XAllocColor(WBGetDefaultDisplay(), colormap, &clrBG);
236  XParseColor(WBGetDefaultDisplay(), colormap, szAFG, &clrAFG);
237  XAllocColor(WBGetDefaultDisplay(), colormap, &clrAFG);
238  XParseColor(WBGetDefaultDisplay(), colormap, szABG, &clrABG);
239  XAllocColor(WBGetDefaultDisplay(), colormap, &clrABG);
240 
241  iInitColorFlag = 1;
242  }
243 }

246 WBEditWindow *WBCreateEditWindow(WBFrameWindow *pOwner, XFontStruct *pFont,
247  const char *szFocusMenu, const WBFWMenuHandler *pHandlerArray,
248  int fFlags)
249 {
250 WBEditWindow *pRval;
251 
252 
253  InternalCheckEWColorsAndAtoms();
254 
255  pRval = (WBEditWindow *)WBAlloc(sizeof(*pRval));
256 
257  if(!pRval)
258  {
259  WB_ERROR_PRINT("ERROR: %s - not enough memory\n", __FUNCTION__);
260  return NULL;
261  }
262 
263  bzero(pRval, sizeof(*pRval));
264 
265  pRval->ulTag = EDIT_WINDOW_TAG;
266  pRval->szFileName = NULL; // explicitly do this, though the bzero would've
267  pRval->pUserCallback = NULL; // explicitly do this, too
268 
269  if(!pFont)
270  {
271  pFont = WBGetDefaultFont();
272  }
273 
274  WBInitializeInPlaceTextObject(&(pRval->xTextObject), None);
275 
276 // pRval->xTextObject.vtable->set_col(&(pRval->xTextObject), 0);
277 // pRval->xTextObject.vtable->set_row(&(pRval->xTextObject), 0);
278 
279 
280  // create the actual window.
281 
282  if(0 > FWInitChildFrame(&(pRval->childframe), pOwner, pFont, // NOTE: a copy of pFont will be in 'childframe.pFont'
283  szFocusMenu, pHandlerArray,
284  FWEditWindowEvent, fFlags))
285  {
286  WB_ERROR_PRINT("ERROR: %s - unable to initialize child frame\n", __FUNCTION__);
287 
288  WBFree(pRval);
289 
290  return NULL;
291  }
292 
293  pRval->xTextObject.wIDOwner = pRval->childframe.wID; // TODO: make assigning this an API function?
294 
295 
296  // assign my 'UI' vtable pointer, which will be (intelligently) called by the 'Child Frame' event handler
297  // this standardizes the various UI methods and makes coding a complex UI quite a bit easier
298  pRval->childframe.pUI = &internal_CFUI; // UI function vtable
299 
300  // assign my 'destructor', which will be called by FWDestroyChildFrame
301  // THIS must be done LAST, since 'FWInitChildFrame' might call FWDestroyChildFrame on error
302  // and I don't want to call the 'destructor' yet.
303  pRval->childframe.destructor = InternalEditWindowDestructor;
304 
305 
306  // TODO: any other initialization belongs HERE. if error, call FWDestroyChildFrame() and return
307  // a NULL immediately (since that would destroy the child frame AND the Edit Window stuff)
308 
309 
310  internal_update_status_text(pRval); // update status text now.
311 
312  CreateTimer(WBGetWindowDisplay(pRval->childframe.wID), pRval->childframe.wID,
313  333333, 1, 1); // TODO: use #define for timer ID and period (1/3 second for now))
314  // NOTE: when I unregister the window callback, the timer will be deleted automatically
315 
316  return pRval;
317 }

373  {
374  WB_ERROR_PRINT("ERROR: %s - destructor is NULL - pointer will not be free'd\n", __FUNCTION__);

387  // the last step destroys the child frame, which will destroy the Edit Window as well

389  FWDestroyChildFrame(&(pEditWindow->childframe)); // destroy window, free up all resources, call destructor (if not NULL)

484 int FWEditWindowEvent(Window wID, XEvent *pEvent)
485 {
486 WBEditWindow *pE;
487 GC gc;
488 WB_GEOM geom, geom2;
489 Display *pDisplay = WBGetWindowDisplay(wID);
490 XFontSet xFontSet;
491 
492 
493  pE = WBEditWindowFromWindowID(wID);
494  if(!pE)
495  {
496  return 0;
497  }
498 
499  switch(pEvent->type)
500  {
501  case Expose:
502 // WB_ERROR_PRINT("TEMPORARY: %s - expose event\n", __FUNCTION__);
503  geom.x = pEvent->xexpose.x;
504  geom.y = pEvent->xexpose.y;
505  geom.width = pEvent->xexpose.width;
506  geom.height = pEvent->xexpose.height;
507 
508  // TEMPORARY - just erase the background, for now...
509  gc = WBBeginPaintGeom(wID, &geom);
510 
511  XSetForeground(pDisplay, gc, clrBG.pixel);
512  XSetBackground(pDisplay, gc, clrBG.pixel);
513 
514  WBGetWindowGeom(wID, &geom2);
515 
516  geom2.x ++; // 1-pixel border
517  geom2.y ++;
518  geom2.width -= 2; // 1 pixel border on BOTH sides
519  geom2.height -= 2;
520 
521  XFillRectangle(pDisplay, wID, gc, geom.x, geom.y, geom.width, geom.height);
522 
523  xFontSet = WBFontSetFromFont(pDisplay, pE->childframe.pFont);
524 
525  pE->xTextObject.vtable->do_expose(&(pE->xTextObject), pDisplay, wID, gc,
526  &geom, // the GEOM to 'paint to'
527  &geom2, // the GEOM bordering the window's viewport (NULL for ALL)
528  xFontSet);
529  WBEndPaint(wID, gc);
530 
531  if(xFontSet)
532  {
533  XFreeFontSet(pDisplay, xFontSet);
534  }
535 
536  return 1; // "handled"
537 
538  case DestroyNotify:
539  // if I'm destroying ME, then I must free up the structure.
540  // if this callback is being called, assume NOT recursive.
541 
542  if(pEvent->xdestroywindow.window == wID)
543  {
544  pE->childframe.wID = None; // assign 'none' as the window ID, since I already destroyed it. don't re-destroy it.
545 
546  WBDestroyEditWindow(pE); // this should fix everything else.
547 
548  return 1; // handled
549  }
550 
551  break;
552 
553  case ClientMessage:
554  if(pEvent->xclient.message_type == aRESIZE_NOTIFY)
555  {
556  // TODO: process re-calculation of the extents, etc.
557 
558 
559  FWSetChildFrameScrollInfo(&(pE->childframe),
560  pE->xTextObject.vtable->get_row(&(pE->xTextObject)),
561  pE->xTextObject.vtable->get_rows(&(pE->xTextObject)),
562  pE->xTextObject.vtable->get_col(&(pE->xTextObject)),
563  pE->xTextObject.vtable->get_cols(&(pE->xTextObject)),
564  pE->childframe.pFont->ascent + pE->childframe.pFont->ascent + EDIT_WINDOW_LINE_SPACING,
565  WBFontAvgCharWidth(pDisplay, pE->childframe.pFont));
566  }
567  else if(pEvent->xclient.message_type == aWM_TIMER)
568  {
569  // only when this tab is visible do I call the callback.
570 
571  if(pE->childframe.pOwner && // just in case
572  FWGetChildFrameIndex(pE->childframe.pOwner, NULL) // focus window's tab index
573  == FWGetChildFrameIndex(pE->childframe.pOwner, &(pE->childframe))) // THIS window's tab index
574  {
