141 static int MessageBoxCallback(Window wID, XEvent *pEvent)
142 {
143 WBDialogWindow *pDlg = DLGGetDialogWindowStruct(wID);
144 struct _MESSAGE_BOX_ *pUserData = (struct _MESSAGE_BOX_ *)(pDlg ? pDlg->pUserData : NULL);
145 
146 
147  if(pEvent->type == ClientMessage && pEvent->xclient.message_type == aDIALOG_INIT)
148  {
149  if(!pDlg)
150  {
151  WB_ERROR_PRINT("MessageBoxCallback - no WBDialogWindow structure in DIALOG_INIT for %d (%08xH) %p %08xH %08xH\n",
152  (unsigned int)wID, (unsigned int)wID, WBGetWindowData(wID, 0), DIALOG_WINDOW_TAG, ((WBDialogWindow *)WBGetWindowData(wID, 0))->ulTag);
153  return 0; // can't process any messages now
154  }
155  else
156  {
157  // assigning the correct icon
158 
159  Window wIDIcon = DLGGetDialogControl(pDlg, 1000); // ID 1000 for icon
160  WBDialogControl *pCtrl = DLGGetDialogControlStruct(wIDIcon);
161 
162  if(pCtrl)
163  {
164  Pixmap pixmap2 = None;
165  Pixmap pixmap = PXM_GetIconPixmap(GetMessageBoxIconPixmapID(pUserData->iType & MessageBox_ICON_MASK),
166  NULL, &pixmap2);
167 
168  if(pixmap != None)
169  WBDialogControlSetIconPixmap(pCtrl, pixmap, pixmap2);
170  }
171  }
172 
173  // assign the caption text to the caption window (which varies and must be assigned at run time)
174 
175  DLGSetControlCaption((WBDialogWindow *)pDlg, 1001, pUserData->szMessage);
176 
177  return 1;
178  }
179 
180  if(!pDlg)
181  {
182  WB_WARN_PRINT("MessageBoxCallback - no WBDialogWindow structure\n");
183  return 0; // can't process any messages now
184  }
185 
186  if(pEvent->type == ClientMessage && pEvent->xclient.message_type == aCONTROL_NOTIFY)
187  {
188  WB_DEBUG_PRINT(DebugLevel_Light | DebugSubSystem_Event | DebugSubSystem_Dialog,
189  "%s - MessageBox ClientMessage CONTROL_NOTIFY\n", __FUNCTION__);
190 
191  switch(pEvent->xclient.data.l[1]) // control ID
192  {
193  case IDOK:
194  case IDCANCEL:
195  if(pEvent->xclient.data.l[0] == aBUTTON_PRESS)
196  {
197  WBEndModal(wID, pEvent->xclient.data.l[1]);
198  }
199  break;
200 
201  default:
202  WB_WARN_PRINT("%s - MessageBox ClientMessage CONTROL_NOTIFY client id=%lx\n",
203  __FUNCTION__, pEvent->xclient.data.l[1]);
204  }
205  }
206

212  return 0;

355  {
356  // assigning the correct icon
357  {
358  WBDialogControl *pCtrl = DLGGetDialogControlStructFromID(pDlg, 1000); // ID 1000 for icon
359 
360  if(pCtrl)

365  if(pixmap != None)
366  {
367  WBDialogControlSetIconPixmap(pCtrl, pixmap, pixmap2);
368  }
369  }
370  }
371 
372  // assign the caption text to the caption window (which varies and must be assigned at run time)
373 
374  DLGSetControlCaption((WBDialogWindow *)pDlg, 1001, pUserData->szMessage);
375 
376  if(pUserData->szRval)
377  {
378  DLGSetControlCaption(pDlg, 1002, pUserData->szRval);
379  }
380 
381  return 1;
382  }
383 
384  if(pEvent->type == ClientMessage && pEvent->xclient.message_type == aCONTROL_NOTIFY)
385  {
386  WB_DEBUG_PRINT(DebugLevel_Light | DebugSubSystem_Event | DebugSubSystem_Dialog,
387  "%s - MessageBox ClientMessage CONTROL_NOTIFY\n", __FUNCTION__);
388

390  {
391  case IDOK:
392  case IDCANCEL:
393  if(pEvent->xclient.data.l[0] == aBUTTON_PRESS)
394  {
395  const char *pText = DLGGetControlCaption(pDlg, 1002);

417  "%s - INPUT BOX 'EDIT' NOTIFICATION %ld\n",
418  __FUNCTION__, pEvent->xclient.data.l[0]);
419 
420  // TODO: update window text

422 
423  default:
424  WB_WARN_PRINT("%s - MessageBox ClientMessage CONTROL_NOTIFY client id=%lx\n",
425  __FUNCTION__, pEvent->xclient.data.l[0]);
426  }
427  }
428 
429  return 0;
430 }

526 static int FileDialogCallback(Window wID, XEvent *pEvent)
527 {
528 WBDialogWindow *pDlg = DLGGetDialogWindowStruct(wID);
529 struct _FILE_DIALOG_ *pUserData = (struct _FILE_DIALOG_ *)(pDlg ? pDlg->pUserData : NULL);
530 Display *pDisplay = WBGetWindowDisplay(wID);
531 char *p1, *p2;
532 
533 
534  if(!pDlg)
535  return 0; // can't process any messages now
536 
537  if(pEvent->type == ClientMessage && pEvent->xclient.message_type == aDIALOG_INIT)
538  {
539  if(!pUserData->szPathName || !*(pUserData->szPathName) ||
540  (!strchr(pUserData->szPathName, '/') && !WBIsDirectory(pUserData->szPathName)))
541  {
542  if(!pUserData->szDefPath || !*(pUserData->szDefPath))
543  {
544  // set the 'DLGC_PATH' property for appropriate controls
545  DLGSetControlProperty(pDlg, FILE_DIALOG_FILE_LIST_CONTROL,
546  aDLGC_PATH, "."); // set path to '.' if not specified already
547  }
548  else
549  {
550  DLGSetControlProperty(pDlg, FILE_DIALOG_FILE_LIST_CONTROL,
551  aDLGC_PATH, pUserData->szDefPath);
552  }
553  }
554  else
555  {
556  p1 = WBCopyString(pUserData->szDefPath);
557  if(!p1)
558  {
559  WB_ERROR_PRINT("%s - no memory to copy path string (a)\n", __FUNCTION__);
560  }
561  else
562  {
563  p2 = strrchr(p1, '/');
564  if(p2)
565  {
566  p2[1] = 0; // keep the '/'
567  }
568 
569  DLGSetControlProperty(pDlg, FILE_DIALOG_FILE_LIST_CONTROL,
570  aDLGC_PATH, p1);
571 
572  free(p1);
573  }
574  }
575 
576  return 1;
577  }
578 
579  if(pEvent->type != ClientMessage)
580  {
581  return 0; // unhandled at this time
582  }
583 
584  // EVERYTHING AT THIS POINT IS A CLIENT MESSAGE
585 
586  if(pEvent->xclient.message_type == aCONTROL_NOTIFY)
587  {
588  // l[0] == message
589  // l[1] == control ID
590  // l[2] ==
591  switch(pEvent->xclient.data.l[1])
592  {
593  case IDOK:
594  case IDCANCEL:
595  if(pEvent->xclient.data.l[0] == aBUTTON_PRESS)
596  {
597  if(pEvent->xclient.data.l[1] == IDOK)
598  {
599  const char *pPath = DLGGetControlCaption(pDlg, FILE_DIALOG_FILE_NAME_CONTROL);
600  const char *pDir = DLGGetControlProperty(pDlg, FILE_DIALOG_FILE_LIST_CONTROL, aDLGC_PATH);
601 
602  if(pUserData->szPathName)
603  {
604  free(pUserData->szPathName);
605  }
606 
607  if(pPath && pPath[0] == '/')
608  {
609  pUserData->szPathName = WBCopyString(pPath); // absolute path assignment
610  }
611  else
612  {
613  if(pDir && *pDir)
614  {
615  pUserData->szPathName = WBCopyString(pDir);
616 
617  if(pUserData->szPathName && *pUserData->szPathName &&
618  pUserData->szPathName[strlen(pUserData->szPathName) - 1] != '/')
619  {
620  WBCatString(&(pUserData->szPathName), "/");
621  }
622  }
623  else
624  {
625  pUserData->szPathName = WBCopyString("./");
626  }
627 
628 // WB_ERROR_PRINT("TEMPORARY %s ---> dir=\"%s\" path=\"%s\" \n", __FUNCTION__, pUserData->szPathName, pPath);
629 
630  if(pUserData->szPathName && pPath)
631  {
632  WBCatString(&(pUserData->szPathName), pPath);
633  }
634 
635  // check for '/../' within path - if present, canonicalize it
636  if(pUserData->szPathName &&
637  (strstr(pUserData->szPathName, "/../") ||
638  (strlen(pUserData->szPathName) > 3 &&
639  (!memcmp(pUserData->szPathName, "../", 3) ||
640  !memcmp(pUserData->szPathName + strlen(pUserData->szPathName) - 3, "/..", 3)))))
641  {
642  p1 = WBGetCanonicalPath(pUserData->szPathName);
643  if(p1)
644  {
645  free(pUserData->szPathName);
646  pUserData->szPathName = p1;
647  }
648  else
649  {
650  WB_ERROR_PRINT("ERROR - %s - Unable to get canonical path for \"%s\"\n", __FUNCTION__, pUserData->szPathName);
651  }
652  }
