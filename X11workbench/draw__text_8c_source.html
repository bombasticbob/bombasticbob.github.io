324 // NOTES ON UTF-8 AND INTERNATIONAL CHARACTERS
325 // notes from http://en.wikipedia.org/wiki/UTF-8
326 //
327 // # of bits highest Byte 1 Byte 2 Byte 3 Byte 4 Byte 5 Byte 6
328 // -----------------------------------------------------------------------------
329 // 7 bits U+007F 0xxxxxxx
330 // 11 bits U+07FF 110xxxxx 10xxxxxx
331 // 16 bits U+FFFF 1110xxxx 10xxxxxx 10xxxxxx
332 // 21 bits U+1FFFFF 11110xxx 10xxxxxx 10xxxxxx 10xxxxxx
333 // 26 bits U+3FFFFFF 111110xx 10xxxxxx 10xxxxxx 10xxxxxx 10xxxxxx
334 // 31 bits U+7FFFFFFF 1111110x 10xxxxxx 10xxxxxx 10xxxxxx 10xxxxxx 10xxxxxx
335 //
336 // Because of THIS, it is STILL SAFE to look for <= ' ' for white space, and make some assumptions
337 // about 0-byte, \n, \r, \t, and other control characters (for now, anyway).
338 //
339 // RFC 3629 states "Implementations of the decoding algorithm MUST protect against decoding invalid sequences."
340 // invalid sequences could (in theory) cause burping to the point of security risk.
341 
342 // There are also some "print direction" override characters 202C 202D and 202E .
343 // http://www.unicodemap.org/details/0x202C/index.html (POP Directional Formatting)
344 // http://www.unicodemap.org/details/0x202D/index.html (LTR Override)
345 // http://www.unicodemap.org/details/0x202E/index.html (RTL Override)
346 //
347 // also http://www.unicodemap.org/details/0x200E/index.html (LTR mark)
348 // http://www.unicodemap.org/details/0x200E/index.html (RTL mark)
349 // http://www.unicodemap.org/details/0x202A/index.html (LTR embedding)
350 // http://www.unicodemap.org/details/0x202B/index.html (RTL embedding)
351 //
352 // and these:
353 // http://www.unicodemap.org/details/0x2028/index.html (line separator - alternate LF?)
354 // http://www.unicodemap.org/details/0x202A/index.html (paragraph separator - alternate to CTRL+L ?)
355 //
356 // There are also a handful of various width 'spacing' including NO WIDTH SPACE
357 // 206A through 206F apparently affect "shaping". 205F through 2069 apparently do nothing
358 //
359 // If I pay attention to these, it could put some serious strain on everything. If I don't, it's not really UTF-8
360 // However the focus of THIS project is "only implement what is necessary" (and forget the rest).

363 DT_WORDS * DTGetWordsFromText(Display *pDisplay, WB_FONTC pFont, const char *szText, int iAlignment)
364 {
365  const char *p1, *p2;
366 
367  DT_WORDS *pRval = WBAlloc(sizeof(DT_WORDS) + sizeof(struct __DT_WORD__) * (INITIAL_DT_WORDS_COUNT - 1));
368 
369  if(!pRval)
370  {
371  return NULL;
372  }
373 
374  pRval->nCount = 0;
375  pRval->szText = szText;
376 //#ifdef HAVE_MALLOC_USABLE_SIZE
377  pRval->nMax = (WBAllocUsableSize(pRval) - sizeof(DT_WORDS)) // calculate actual # of structs it contains
378  / sizeof(struct __DT_WORD__)
379  + 1; // because DT_WORDS contains one struct __DT_WORD__
380 //#else // HAVE_MALLOC_USABLE_SIZE
381  pRval->nMax = INITIAL_DT_WORDS_COUNT;
382 //#endif // HAVE_MALLOC_USABLE_SIZE
383 
384  if(!szText || !*szText)
385  {
386  return pRval;
387  }
388 
389  p1 = szText;
390  while(*p1)
391  {
392  // find non-white-space and keep 'tabs' on the count/type of white space as well
393  // adding entries to 'words' as needed
394 
395  while(*p1 && (unsigned char)*p1 <= ' ')
396  {
397  // TODO: check for high bit set aka UTF-8 multi-byte chars
398 
399  if(*p1 == '\t') // tab character
400  {
401  if(!CheckReAllocWords(&pRval))
402  {
403  WBFree(pRval);
404  return NULL; // ERROR
405  }
406 
407  pRval->aWords[pRval->nCount].pText = NULL;
408  pRval->aWords[pRval->nCount].nLength = 1;
409  pRval->aWords[pRval->nCount].iIsWhiteSpace = 0;
410  pRval->aWords[pRval->nCount].iIsLineFeed = 0;
411  pRval->aWords[pRval->nCount].iIsTab = 1;
412  pRval->aWords[pRval->nCount].iX = -1;
413  pRval->aWords[pRval->nCount].iY = -1;
414 
415  while(*(p1 + 1) == '\t')
416  {
417  p1++;
418  pRval->aWords[pRval->nCount].nLength++;
419  }
420 
421  // use white space string for width
422  pRval->aWords[pRval->nCount].iWidth = 0; // NA (calc on the fly)
423  pRval->aWords[pRval->nCount].iHeight = 0; // NA (calc on the fly)
424 
425  pRval->nCount++;
426  }
427  else if(*p1 == '\n' || *p1 == '\r')
428  {
429  if(!CheckReAllocWords(&pRval))
430  {
431  WBFree(pRval);
432  return NULL; // ERROR
433  }
434 
435  pRval->aWords[pRval->nCount].pText = NULL;
436  pRval->aWords[pRval->nCount].nLength = 1;
437 
438  if(iAlignment & DTAlignment_SINGLELINE) // treat line feed as white space
439  {
440  pRval->aWords[pRval->nCount].iIsWhiteSpace = 1;
441  pRval->aWords[pRval->nCount].iIsLineFeed = 0;
442  }
443  else
444  {
445  pRval->aWords[pRval->nCount].iIsWhiteSpace = 0;
446  pRval->aWords[pRval->nCount].iIsLineFeed = 1;
447  }
448 
449  pRval->aWords[pRval->nCount].iIsTab = 0;
450  pRval->aWords[pRval->nCount].iX = -1;
451  pRval->aWords[pRval->nCount].iY = -1;
452 
453  while(1)
454  {
455  if(*p1 == '\n' && p1[1] == '\r')
456  {
457  p1++;
458  }
459  else if(*p1 == '\r' && p1[1] == '\n')
460  {
461  p1++;
462  }
463 
464 // pRval->aWords[pRval->nCount].nLength++;
465 
466  if(p1[1] != '\r' && p1[1] != '\n')
467  {
468  break;
469  }
470 
471  p1++; // loop consecutive CR,LF
472  pRval->aWords[pRval->nCount].nLength++;
473  }
474 
475  // use white space string for width
476  pRval->aWords[pRval->nCount].iWidth = 0; // NA
477  pRval->aWords[pRval->nCount].iHeight = 0; // NA
478 
479  pRval->nCount++;
480  }
481  else if((unsigned char)*p1 <= ' ')
482  {
483  if(!CheckReAllocWords(&pRval))
484  {
485  WBFree(pRval);
486  return NULL; // ERROR
487  }
488 
489  pRval->aWords[pRval->nCount].pText = NULL;
490  pRval->aWords[pRval->nCount].nLength = 1;
491  pRval->aWords[pRval->nCount].iIsWhiteSpace = 1;
492  pRval->aWords[pRval->nCount].iIsLineFeed = 0;
493  pRval->aWords[pRval->nCount].iIsTab = 0;
494  pRval->aWords[pRval->nCount].iX = -1;
495  pRval->aWords[pRval->nCount].iY = -1;
496 
497  p2 = p1;
498 
499  while(*(p1 + 1) && (unsigned char)*(p1 + 1) <= ' ' && *(p1 + 1) != '\t' && *(p1 + 1) != '\r' && *(p1 + 1) != '\n')
500  {
501  p1++;
502  pRval->aWords[pRval->nCount].nLength++;
503  }
504 
505  // use white space string for width
506  pRval->aWords[pRval->nCount].iWidth = DTGetTextWidth(pFont, p2, pRval->aWords[pRval->nCount].nLength);
507  pRval->aWords[pRval->nCount].iHeight = WBFontHeight(pFont); // total height of font
508 
509  pRval->nCount++;
510  }
511 
512  p1++;
513  }
514 
515  if(*p1)
516  {
517  int iUnderscoreFlag = 0;
518 
519  // get length of word
520 
521  p2 = p1;
522  while(/* *p1 && */ (unsigned char)*p1 > ' ')
523  {
524  if(*p1 == '_')
525  {
526  iUnderscoreFlag++;
527  }
528 
529  p1++;
530  }
531 
532  if(!CheckReAllocWords(&pRval))
533  {
534  WBFree(pRval);
535  return NULL; // ERROR
536  }
537 
538  pRval->aWords[pRval->nCount].pText = p2;
539  pRval->aWords[pRval->nCount].nLength = p1 - p2;
540  pRval->aWords[pRval->nCount].iIsWhiteSpace = 0;
541  pRval->aWords[pRval->nCount].iIsTab = 0;
542  pRval->aWords[pRval->nCount].iIsLineFeed = 0;
543  pRval->aWords[pRval->nCount].iX = -1;
544  pRval->aWords[pRval->nCount].iY = -1;
545 
546  if(p1 > p2)
547  {
548  if(!(iAlignment & DTAlignment_UNDERSCORE) || !iUnderscoreFlag)
549  {
550  pRval->aWords[pRval->nCount].iWidth = DTGetTextWidth(pFont, p2, p1 - p2);
551  }
552  else
553  {
554  // fast and dirty, subtract "length of underscores" from text length
555  // assume kerning doesn't alter this at all
556 
557  // TODO: the hard way?
558 
559  pRval->aWords[pRval->nCount].iWidth = DTGetTextWidth(pFont, p2, p1 - p2)
560  - iUnderscoreFlag * DTGetTextWidth(pFont, "_", 1);
561  }
562  }
563  else
564  {
565  pRval->aWords[pRval->nCount].iWidth = 0;
566  }
567 
568  pRval->aWords[pRval->nCount].iHeight = WBFontHeight(pFont);//pFont->ascent + pFont->descent; // for now just do this
569 
570  pRval->nCount++;
571  }
572  }
573 
574  return pRval;
575 }

604 static void DoHCenterWrappedLines(DT_WORDS *pWords, int iWidth)
605 {
606 int i1, i2, i3, iMax;
607 DT_WORD *pW;
608 
609 
610  // For a multi-line string, I need to make sure each line centers
611  // correctly. If the lines break, I allow them to break and wrap
612  // but each line that is COMPLETELY within the centered rectangle
613  // without wrapping MUST BE RE-CENTERED!!!
614 
615  i1 = 0; // this will be my 'word counter'
616 
617  do
618  {
619  i2 = i1; // start of line
620  iMax = 0;
621 
622  // step 1: find the end of the current line and calculate
623  // the max width of the line (including 'wrap')
624 
625  while(i1 < pWords->nCount)
626  {
627  pW = &(pWords->aWords[i1]);
628  i1++;
629 
630  if(pW->iIsLineFeed)
631  {
632  break; // i1 already incremented, so it will skip Mr. Line Feed on the next iteration
633  }
634 
635  if(!pW->iIsTab && !pW->iIsWhiteSpace)
636  {
637  i3 = pW->iX + pW->iWidth; // max extent
638 
639  if(i3 > iMax)
640  {
641  iMax = i3;
642  }
643  }
644  }
645 
646  // if 'iMax' is less than 'iWidth', add half the delta
647  // to all of the 'iX' values
648 
649  i3 = (iWidth - iMax) / 2;
650 
651  if(i3 > 0)
652  {
653  for(; i2 < i1; i2++)
654  {
655  pW = &(pWords->aWords[i2]);
656 
657  if(pW->iX >= 0) // only for the 'assigned' ones
658  {
659  pW->iX += i3;
660  }
661  }
662  }
663 
664  } while(i1 < pWords->nCount);
665 
666 
667 // InternalDebugDumpWords(pWords);
668 }
669 
670 
671 
673 // * parameters to this function
674 // *
675 // * pFont A pointer to an XFontStruct (NULL implies system default font)
676 // * pWords The DT_WORDS structure for the text
677 // * iTabWidth A positive integer in 'characters', or negative integer in pixels, indicating tab width
678 // * iTabOrgin An unsigned integer indicating the tab origin, using the same units as iTabWidth, corresponding to the first character.
