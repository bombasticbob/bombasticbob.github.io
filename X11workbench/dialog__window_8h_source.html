588  long lData0, long lData1, long lData2, long lData3, long lData4)
589 {
590 Display *pDisplay = WBGetWindowDisplay(pDLG->wID);
591 
592  XClientMessageEvent evt = {
593  .type=ClientMessage,
594  .serial=0,
595  .send_event=0,
596  .display=pDisplay,
597  .window=pDLG->wID,
598  .message_type=aNotify,
599  .format=32
600  };
601  evt.data.l[0] = lData0;
602  evt.data.l[1] = lData1;
603  evt.data.l[2] = lData2;
604  evt.data.l[3] = lData3;
605  evt.data.l[4] = lData4;
606 
607  WBWindowDispatch(pDLG->wID, (XEvent *)&evt);
