539  long lData0, long lData1, long lData2, long lData3, long lData4)
540 {

542 
543  XClientMessageEvent evt = {
544  .type=ClientMessage,
545  .serial=0,
546  .send_event=0,
547  .display=pDisplay,
548  .window=pDLG->wID,
549  .message_type=aNotify,
550  .format=32
551  };
552  evt.data.l[0] = lData0;
553  evt.data.l[1] = lData1;
554  evt.data.l[2] = lData2;
555  evt.data.l[3] = lData3;
556  evt.data.l[4] = lData4;
557 
558  WBWindowDispatch(pDLG->wID, (XEvent *)&evt);
