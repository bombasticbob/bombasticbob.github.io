887 static __inline__ void WBInitializeTextObject(TEXT_OBJECT *pTextObject)
888 {
889  if(pTextObject)
890  {
891  memset(pTextObject, 0, sizeof(*pTextObject)); // most objects will accept 'nulling' as "unassigned"
892 
893  pTextObject->ulTag = TEXT_OBJECT_TAG;
894  pTextObject->wIDOwner = None;
895  pTextObject->iLineFeed = LineFeed_DEFAULT;
896 
897  pTextObject->vtable = WBGetDefaultTextObjectVTable();
898  pTextObject->vtable->init(pTextObject);
899  }
900 }
