880 static __inline__ void WBInitializeTextObject(TEXT_OBJECT *pTextObject)
881 {
882  if(pTextObject)
883  {
884  memset(pTextObject, 0, sizeof(*pTextObject)); // most objects will accept 'nulling' as "unassigned"
885 
886  pTextObject->ulTag = TEXT_OBJECT_TAG;
887  pTextObject->wIDOwner = None;
888  pTextObject->iLineFeed = LineFeed_DEFAULT;
889 
890  pTextObject->vtable = WBGetDefaultTextObjectVTable();
891  pTextObject->vtable->init(pTextObject);
892  }
893 }
