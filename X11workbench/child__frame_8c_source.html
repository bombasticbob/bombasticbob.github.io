72 int FWInitChildFrame(WBChildFrame *pChildFrame, WBFrameWindow *pOwner, XFontStruct *pFont,
73  const char *szFocusMenu, const WBFWMenuHandler *pHandlerArray,
74  WBWinEvent pUserCallback, int fFlags)
75 {
76 WBChildFrame *pC;
77 XSetWindowAttributes xswa; /* Temporary Set Window Attribute struct */
78 Display *pDisplay = WBGetDefaultDisplay();
79 int iRval = -1;
80 
81 
82  if(!pChildFrame || !pOwner || pOwner->wID == None)
83  {
84  // TODO: validate 'pOwner' as a valid WBFrameWindow ??
85 
86  WB_ERROR_PRINT("%s - invalid pointers or owner window does not exist\n", __FUNCTION__);
87 
88  return -1;
89  }
90 
91  // zero out entire structure beforehand
92  bzero(pChildFrame, sizeof(*pChildFrame));
93 
94 
95  // FIRST, attach the child frame to my list o' child frames, making sure
96  // that it is not already there [if it is assume this was called twice]
97  // while I'm at it, zero out the structure BEFORE I assign 'pNext'
98 
99  if(!pChildFrames)
100  {
101  pChildFrames = pChildFrame;
102  pChildFrame->pNext = NULL; // make sure
103  }
104  else
105  {
106  pC = pChildFrames;
107 
108  while(pC->pNext)
109  {
110  if(pC == pChildFrame) // just in case
111  {
112  WB_ERROR_PRINT("%s - unexpected error condition, did you call this function before?\n", __FUNCTION__);
113 
114  return -2;
115  }
116 
117  pC = pC->pNext;
118  }
119 
120  if(pC)
121  {
122  // zero out entire structure beforehand
123  bzero(pChildFrame, sizeof(*pChildFrame));
124 
125  pC->pNext = pChildFrame;
126  pChildFrame->pNext = NULL;
127  }
128  else
129  {
130  // internal error, flag it for now and return "fail"
131  WB_ERROR_PRINT("%s - unexpected condition, returning error\n", __FUNCTION__);
132 
133  return -3;
134  }
135  }
136 
137  // if the owner is valid, but NOT tabbed, and it already has
138  // a WBChildFrame, then I fail to create this one.
139 
140  if((WBFrameWindow_NO_TABS & pOwner->iFlags)) // window has NO tabs
141  {
142  if(FWGetNumContWindows(pOwner)) // error *OR* > 0, doesn't matter, can ONLY have one tab in the frame
143  {
144  FWDestroyChildFrame(pChildFrame); // this unhooks everything (alloc'd things are zero'd by bzero)
145 
146  return -4; // can't create, SDI
147  }
148  }
149 
150 
151  // NEXT, set up all of the various 'things' in the structure that are simple assignments
152  pChildFrame->ulTag = CHILD_FRAME_TAG;
153  pChildFrame->destructor = NULL; // make sure
154 
155  if(!pFont)
156  {
157  pChildFrame->pFont = NULL;
158  }
159  else
160  {
161  pChildFrame->pFont = WBCopyFont(pFont);
162  }
163 
164  if(szFocusMenu)
165  {
166  pChildFrame->pszMenuResource = WBCopyString(szFocusMenu);
167 
168  if(!pChildFrame->pszMenuResource)
169  {
170  FWDestroyChildFrame(pChildFrame); // this unhooks everything (alloc'd things are zero'd by bzero)
171  return -5; // not enough memory
172  }
173  }
174  else
175  {
176  pChildFrame->pszMenuResource = NULL;
177  }
178 
179  if(pHandlerArray)
180  {
181  const WBFWMenuHandler *pH;
182  int i1;
183 
184  // count the number of entries in the menu handler
185  for(i1=0, pH = pHandlerArray; pH->lMenuID || pH->callback || pH->UIcallback; i1++, pH++)
186  {
187  // NOTHING inside the loop. just count.
188  }
189 
190  // allocate space and make a copy
191 
192  pChildFrame->pMenuHandler = (WBFWMenuHandler *)WBAlloc(sizeof(WBFWMenuHandler) * (i1 + 2));
193 
194  if(pChildFrame->pMenuHandler)
195  {
196  memcpy(pChildFrame->pMenuHandler, pHandlerArray,
197  sizeof(WBFWMenuHandler) * (i1 + 1));
198  }
199  else
200  {
201  FWDestroyChildFrame(pChildFrame); // this unhooks everything (alloc'd things are zero'd by bzero)
202  return -5; // not enough memory
203  }
204  }
205 
206  // Next, I need to create the window via 'WBCreateWindow()'
207 
208  bzero(&xswa, sizeof(xswa));
209 
210  xswa.border_pixel = FWGetDefaultBD().pixel;
211  xswa.background_pixel = FWGetDefaultBG().pixel; // typically a 'grey' color
212  xswa.colormap = DefaultColormap(pDisplay, DefaultScreen(pDisplay));
213  xswa.bit_gravity = CenterGravity;
214 
215  pChildFrame->wID = WBCreateWindow(pDisplay, pOwner->wID, FWChildFrameEvent, "ChildFrame",
216  pOwner->iClientX, pOwner->iClientY,
217  pOwner->iClientWidth - 2, pOwner->iClientHeight - 2, // border is 1 pixel, so subtract 2
218  1, InputOutput,
219  CWBorderPixel | CWBackPixel | CWColormap | CWBitGravity,
220  &xswa);
221 
222  if(pChildFrame->wID == None)
223  {
224  FWDestroyChildFrame(pChildFrame);
225  return -6; // could not create the window
226  }
227 
228  // immediately identify this window's data struct using window data
229  WBSetWindowData(pChildFrame->wID, 0, (void *)pChildFrame); // window data is pointer to this struct
230  pChildFrame->pUserCallback = pUserCallback;
231 
232 
233  // create the default graphics context using the frame's color info
234  WBCreateWindowDefaultGC(pChildFrame->wID, FWGetDefaultFG().pixel, FWGetDefaultBG().pixel);
235 
236  // now allow specific kinds of input messages. I won't need 'structure notify' as they won't work
237  // anyway. those end up going to the frame window, and it passes them (as needed) to me.
238  XSelectInput(pDisplay, pChildFrame->wID,
239  WB_STANDARD_INPUT_MASK | WB_MOUSE_INPUT_MASK | WB_KEYBOARD_INPUT_MASK);
240 
241 
242  // FINALLY, inform the frame window, which will set up some other stuff for me
243 
244  pChildFrame->pOwner = pOwner;
245  iRval = FWAddContainedWindow(pOwner, pChildFrame); // TODO: make THIS assign the owner pointer?
246 
247  if(iRval < 0) // error? (returns negative on error, otherwise the tab index)
248  {
249  FWDestroyChildFrame(pChildFrame); // this frees up all of the resources (but does not free the mem block)
250  }
251 
252  // TODO: other things
253 
254  return iRval;
255 }

257 void FWDestroyChildFrame(WBChildFrame *pChildFrame)
258 {
259 WBChildFrame *pC, *pC2;
260 Window wID;
261 
262 
263  if(!pChildFrame)
264  {
265  WB_ERROR_PRINT("ERROR: %s - pChildFrame not valid\n", __FUNCTION__);
266 
267  return;
268  }
269 
270  wID = pChildFrame->wID; // make a copy of it
271 
272  // FIRST, tell the owner to remove me from the list
273 
274  if(pChildFrame->pOwner)
275  {
276  FWRemoveContainedWindow(pChildFrame->pOwner, pChildFrame);
277 
278  if(pChildFrame->pOwner != NULL)
279  {
280  WB_ERROR_PRINT("ERROR: %s - owner pointer was nut 'NULL'd out\n", __FUNCTION__);
281  }
282 
283  pChildFrame->pOwner = NULL;
284  }
285 
286  // NOW, free up any resources that were WBAlloc'd - they must also be free'd
287 
288  if(pChildFrame->szDisplayName)
289  {
290  WBFree(pChildFrame->szDisplayName);
291  pChildFrame->szDisplayName = NULL;
292  }
293 
294  if(pChildFrame->pFont)
295  {
296  XFreeFont(WBGetDefaultDisplay(), pChildFrame->pFont);
297  pChildFrame->pFont = NULL;
298  }
299 
300  if(pChildFrame->pMenuHandler)
301  {
302  WBFree(pChildFrame->pMenuHandler);
303  pChildFrame->pMenuHandler = NULL;
304  }
305 
306  if(pChildFrame->szStatusText)
307  {
308  WBFree(pChildFrame->szStatusText);
309  pChildFrame->szStatusText = NULL;
310  }
311 
312  if(pChildFrame->pszMenuResource)
313  {
314  WBFree(pChildFrame->pszMenuResource);
315  pChildFrame->pszMenuResource = NULL;
316  }
317 
318  // And, NOW, destroy the window itself [if it's valid]
319  //
320  // NOTE: to prevent recursion problems, a 'DestroyNotify' would mark the window ID 'None'
321  // before calling this function. this will happen within my OWN proc, if the superclass doesn't do it
322 
323  if(pChildFrame->wID != None) // if there's a chance of recursion, I detect it here
324  {
325  WBDestroyWindow(pChildFrame->wID);
326  pChildFrame->wID = None; // make sure
327  }
328 
329  // FINALLY, unlink this object from the child frame list. Do it by walking the list, looking for
330  // this entry in the list. If I find it, I unlink it [otherwise, the 'pNext' link might be *stale*]
331 
332  if(pChildFrames) // just in case, walk the list - do not rely on possibly stale pointers
333  {
334  pC2 = NULL;
335  pC = pChildFrames;
336 
337  while(pC)
338  {
339  if(pC == pChildFrame) // just in case
340  {
341  if(pC2)
342  {
343  pC2->pNext = pC->pNext; // remove from link list
344  }
345  else
346  {
347  pChildFrames = pC->pNext; // it was the head of the chain
348  }
349 
350  pC->pNext = NULL;
351 
352  break;
353  }
354 
355  pC2 = pC;
356  pC = pC->pNext;
357  }
358  }
359 
360  pChildFrame->pNext = NULL; // make sure it's not a pointer someplace that can cause problems
361 
362  // if the 'superclass' destructor was specified, call it NOW.
363 
364  if(pChildFrame->destructor)
365  {
366  // at this point, we ALWAYS destroy the callback function.
367  // "it is assumed" that I won't be needing 'DestroyNotify' to
368  // trigger any destruction. That happens *AFTER* the window was
369  // destroyed, and happens asynchronously, typically before the
370  // frame window was destroyed.
371 
372  // so, to make sure that I don't get an event callback that references
373  // the window object after it's free'd, I need to remove the callback.
374 
375  if(wID != None)
376  {
377  WBUnregisterWindowCallback(wID); // no more event handling
378  WBSetWindowData(wID, 0, (void *)NULL); // window data is no longer pointing here
379  }
380 
381  // and THEN, it's safe to call the destructor on the original object.
382  pChildFrame->destructor(pChildFrame); // this should auto-delete as well
383  }
384  else
385  {
386  // TODO: should I perform a 'Destroy Notify' callback of my own?
387 
388  // NOTE: NOT calling 'WBunregisterWindowCallback()' so 'Destroy Notify' does its job. hopefully.
389  }
390 
391  // NOTE: I have to leave the tag alone at this point, to validate that it's
392  // a 'Child Frame' structure. But everything else needs to be zero'd out
393  // (I used to use 'bzero' but I'm just commenting it here, for reference)
394  // so... make sure everything that's free'd gets ZERO'd as well. this is
395  // because a 'DestroyNotify' handler that frees up the memory (and destroys
396  // any 'superclass') will probably call this function again, and it might
397  // happen asynchronously. In fact, it probably WILL happen asynchronously.
398 
399  // bzero(pChildFrame, sizeof(*pChildFrame)); // zero it out (as a matter of course)
400 
401  // note that I do *NOT* free the WBChildFrame HERE - this simply releases all resources for it.
402  // If there is NO DESTRUCTOR, the 'Destroy Notify' handler needs to do that, or at least SOMETHING.
403  // *NOT* handling this properly can result in 'use after free' problems.
404 }

545 void FWChildFrameRecalcLayout(WBChildFrame *pChildFrame)
546 {
547 WBFrameWindow *pOwner;
548 Display *pDisplay;
549 int iL, iT, iW, iH;
550 
551 
552  if(!pChildFrame || pChildFrame->ulTag != CHILD_FRAME_TAG)
553  {
554  WB_ERROR_PRINT("ERROR: %s - pChildFrame not valid\n", __FUNCTION__);
555 
556  return;
557  }
558 
559 
560  pDisplay = WBGetWindowDisplay(pChildFrame->wID);
561  pOwner = pChildFrame->pOwner;
562 
563  if(!pOwner) // irrelevant if NULL. TODO: properly validate this (anal retentive if DEBUG build)
564  {
565  WB_ERROR_PRINT("ERROR: %s - pOwner is NULL for Child Frame\n", __FUNCTION__);
566 
567  return;
568  }
569 
570  iL = pOwner->iClientX;
571  iT = pOwner->iClientY;
572  iW = pOwner->iClientWidth;
573  iH = pOwner->iClientHeight;
574 
575  // resize the window accordingly
576  XMoveWindow(pDisplay, pChildFrame->wID, iL, iT);
577  XResizeWindow(pDisplay, pChildFrame->wID, iW - 2, iH - 2); // allow 1 pixel for border
578 
579  // calculate new client 'geom', backing out 2 additional pixels in all 4 directions
580 
581  pChildFrame->geom.x = 2;
582  pChildFrame->geom.y = 2;
583  pChildFrame->geom.width = iW - 6;
584  pChildFrame->geom.height = iH - 6; // the new width/height of client area
585 
586 
587  // TODO: anything ELSE that I need to do when re-calculating the layout, scrollbars, whatever
588  // just apply that to geom so that it reflects the correct viewpoirt in pixels
589  // minus any border, decorations, scrollbars, whatever, with 0,0 being top,left
590  // for the window [in this case, border is 1 pixel already]
591 
592 
593 
594  // TODO: if I need to show scrollbars, subtract height/width of bar (plus border) from
595  // the geometry width and height, as needed
596 
597 
598 // pChildFrame->iLeft = iL converted to correct units and scrolled;
599 // pChildFrame->iTop = iT converted to correct units and scrolled;
600 // pChildRrame->iWidth = iW converted to correct units;
601 // pChildFrame->iHeight = iH converted to correct units;
602 
603 
604 
605  // NOW, tell the user callback function (if any) what's happening.
606  // I must assume that the owning frame is valid and has already re-calc'd its layout
607 
608  if(pChildFrame->pUserCallback)
609  {
610  Display *pDisplay;
611  XClientMessageEvent evt;
612 
613  pDisplay = WBGetWindowDisplay(pChildFrame->wID);
614 
615  bzero(&evt, sizeof(evt));
616  evt.type = ClientMessage;
617 
618  evt.display = pDisplay;
619  evt.window = pChildFrame->wID;
620  evt.message_type = aRESIZE_NOTIFY;
621  evt.format = 32; // always
622  evt.data.l[0] = pChildFrame->geom.x; // left
623  evt.data.l[1] = pChildFrame->geom.y; // top
624  evt.data.l[2] = pChildFrame->geom.x + pChildFrame->geom.width; // right
625  evt.data.l[3] = pChildFrame->geom.y + pChildFrame->geom.height; // bottom
626 
627  // required implementation. superclass must process this and fix up scroll info
628  pChildFrame->pUserCallback(evt.window, (XEvent *)&evt);
629  }
630 
631 
632  // TODO: fix the scrollbars and invalidate rectangles
633 
634 }

686 int FWChildFrameEvent(Window wID, XEvent *pEvent)
687 {
688 int iRval = 0;
689 WBChildFrame *pC;
690 WBFrameWindow *pFW;
691 #ifndef NO_DEBUG
692 char tbuf[32]; // for keyboard input
693 int nChar = sizeof(tbuf);
694 #endif // NO_DEBUG

696 
697  pC = FWGetChildFrameStruct(wID);
698 
699  if(!pC)

935  }
936 
937  break;
938 
939  case WB_POINTER_CANCEL:
940  if(pUI->mouse_cancel)
941  {
942  pUI->mouse_cancel(pC);

971 
972  break;
973  }
974
