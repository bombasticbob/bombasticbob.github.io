64 int FWInitChildFrame(WBChildFrame *pChildFrame, WBFrameWindow *pOwner, XFontStruct *pFont,
65  const char *szFocusMenu, const WBFWMenuHandler *pHandlerArray,
66  WBWinEvent pUserCallback, int fFlags)
67 {
68 WBChildFrame *pC;
69 XSetWindowAttributes xswa; /* Temporary Set Window Attribute struct */
70 Display *pDisplay = WBGetDefaultDisplay();
71 int iRval = -1;
72 
73 
74  if(!pChildFrame || !pOwner || pOwner->wID == None)
75  {
76  // TODO: validate 'pOwner' as a valid WBFrameWindow ??
77 
78  WB_ERROR_PRINT("%s - invalid pointers or owner window does not exist\n", __FUNCTION__);
79 
80  return -1;
81  }
82 
83  // FIRST, attach the child frame to my list o' child frames, making sure
84  // that it is not already there [if it is assume this was called twice]
85  // while I'm at it, zero out the structure BEFORE I assign 'pNext'
86 
87  if(!pChildFrames)
88  {
89  // zero out entire structure beforehand
90  bzero(pChildFrame, sizeof(*pChildFrame));
91 
92  pChildFrames = pChildFrame;
93  pChildFrame->pNext = NULL; // make sure
94  }
95  else
96  {
97  pC = pChildFrames;
98 
99  while(pC->pNext)
100  {
101  if(pC == pChildFrame) // just in case
102  {
103  WB_ERROR_PRINT("%s - unexpected error condition, did you call this function before?\n", __FUNCTION__);
104 
105  return -2;
106  }
107 
108  pC = pC->pNext;
109  }
110 
111  if(pC)
112  {
113  // zero out entire structure beforehand
114  bzero(pChildFrame, sizeof(*pChildFrame));
115 
116  pC->pNext = pChildFrame;
117  pChildFrame->pNext = NULL;
118  }
119  else
120  {
121  // internal error, flag it for now and return "fail"
122  WB_ERROR_PRINT("%s - unexpected condition, returning error\n", __FUNCTION__);
123 
124  return -3;
125  }
126  }
127 
128  // if the owner is valid, but NOT tabbed, and it already has
129  // a WBChildFrame, then I fail to create this one.
130 
131  if((WBFrameWindow_NO_TABS & pOwner->iFlags)) // window has NO tabs
132  {
133  if(FWGetNumContWindows(pOwner)) // error *OR* > 0, doesn't matter
134  {
135  FWDestroyChildFrame(pChildFrame); // this unhooks everything (alloc'd things are zero'd by bzero)
136 
137  return -4; // can't create, SDI
138  }
139  }
140 
141 
142  // NEXT, set up all of the various 'things' in the structure that are simple assignments
143  pChildFrame->ulTag = CHILD_FRAME_TAG;
144 
145  if(!pFont)
146  {
147  pChildFrame->pFont = NULL;
148  }
149  else
150  {
151  pChildFrame->pFont = WBCopyFont(pFont);
152  }
153 
154 
155  // Next, I need to create the window via 'WBCreateWindow()'
156 
157  bzero(&xswa, sizeof(xswa));
158 
159  xswa.border_pixel = FWGetDefaultBD().pixel;
160  xswa.background_pixel = FWGetDefaultBG().pixel;
161  xswa.colormap = DefaultColormap(pDisplay, DefaultScreen(pDisplay));
162  xswa.bit_gravity = CenterGravity;
163 
164  pChildFrame->wID = WBCreateWindow(pDisplay, pOwner->wID, FWChildFrameEvent, "ChildFrame",
165  pOwner->iClientX, pOwner->iClientY,
166  pOwner->iClientWidth, pOwner->iClientHeight,
167  1, InputOutput,
168  CWBorderPixel | CWBackPixel | CWColormap | CWBitGravity,
169  &xswa);
170 
171  if(pChildFrame->wID == None)
172  {
173  FWDestroyChildFrame(pChildFrame);
174  return -6; // could not create the window
175  }
176 
177  // immediately identify this window's data struct using window data
178  WBSetWindowData(pChildFrame->wID, 0, (void *)pChildFrame); // window data is pointer to this struct
179  pChildFrame->pUserCallback = pUserCallback;
180 
181 
182  // create the default graphics context using the frame's color info
183  WBCreateWindowDefaultGC(pChildFrame->wID, FWGetDefaultFG().pixel, FWGetDefaultBG().pixel);
184 
185  // now allow specific kinds of input messages
186  XSelectInput(pDisplay, pChildFrame->wID,
187  WB_STANDARD_INPUT_MASK | WB_MOUSE_INPUT_MASK | WB_KEYBOARD_INPUT_MASK);
188 
189 
190  // FINALLY, inform the frame window, which will set up some other stuff for me
191 
192  pChildFrame->pOwner = pOwner;
193  iRval = FWAddContainedWindow(pOwner, pChildFrame); // TODO: make THIS assign the owner pointer?
194 
195  if(iRval) // error?
196  {
197  FWDestroyChildFrame(pChildFrame); // this frees up all of the resources
198  }
199 
200  // TODO: other things
201 
202  return iRval;
203 }

205 void FWDestroyChildFrame(WBChildFrame *pChildFrame)
206 {
207 WBChildFrame *pC, *pC2;
208 
209 
210  if(!pChildFrame)
211  {
212  return;
213  }
214 
215  // FIRST, unlink this object from the child frame list
216 
217  if(pChildFrames) // just in case
218  {
219  pC2 = NULL;
220  pC = pChildFrames;
221 
222  while(pC)
223  {
224  if(pC == pChildFrame) // just in case
225  {
226  if(pC2)
227  {
228  pC2->pNext = pC->pNext; // remove from link list
229  }
230  else
231  {
232  pChildFrames = pC->pNext; // it was the head of the chain
233  }
234 
235  pC->pNext = NULL;
236 
237  break;
238  }
239 
240  pC2 = pC;
241  pC = pC->pNext;
242  }
243  }
244 
245  // next, tell the owner to remove me from the list
246 
247  if(pChildFrame->pOwner)
248  {
249  FWRemoveContainedWindow(pChildFrame->pOwner, pChildFrame);
250  }
251 
252 
253  // FINALLY, free the window
254  WBDestroyWindow(pChildFrame->wID);
255 
256  // and any resources that were malloc'd must also be free'd
257 
258  if(pChildFrame->szDisplayName)
259  {
260  free(pChildFrame->szDisplayName);
261  }
262 
263  if(pChildFrame->pFont)
264  {
265  XFreeFont(WBGetDefaultDisplay(), pChildFrame->pFont);
266  }
267 
268  bzero(pChildFrame, sizeof(*pChildFrame)); // zero it out [so I don't re-free things if I call something twice]
269 
270 }

290 void FWChildFrameRecalcLayout(WBChildFrame *pChildFrame)
291 {
292 WBFrameWindow *pOwner;
293 int iL, iT, iW, iH;
294 
295 
296  if(!pChildFrame) // TODO: properly validate this (anal retentive if DEBUG build)
297  {
298  return;
299  }
300 
301  pOwner = pChildFrame->pOwner;
302 
303  if(!pOwner) // irrelevant if NULL. TODO: properly validate this (anal retentive if DEBUG build)
304  {
305  return;
306  }
307 
308  iL = pOwner->iClientX;
309  iT = pOwner->iClientY;
310  iW = pOwner->iClientWidth;
311  iH = pOwner->iClientHeight;
312 
313 // re-calculate these members of the WBChildFrame structure, as needed
314 
315 // pChildFrame->iLeft = iL converted to correct units and scrolled;
316 // pChildFrame->iTop = iT converted to correct units and scrolled;
317 // pChildRrame->iWidth = iW converted to correct units;
318 // pChildFrame->iHeight = iH converted to correct units;
319 
320 
321 
322 
323  // TODO: anything ELSE that I need to do when re-calculating the layout, scrollbars, whatever
324  // just apply that to iL, iT, iW, iH so that it reflects the correct viewpoirt in pixels
325  // minus any border, decorations, scrollbars, whatever, with 0,0 being top,left
326 
327 
328  // NOW, tell the user callback function (if any) what's happening.
329  // I must assume that the owning frame is valid and has already re-calc'd its layout
330 
331  if(pChildFrame->pUserCallback)
332  {
333  Display *pDisplay;
334  XClientMessageEvent evt;
335 
336  pDisplay = WBGetWindowDisplay(pChildFrame->wID);
337 
338  bzero(&evt, sizeof(evt));
339  evt.type = ClientMessage;
340 
341  evt.display = pDisplay;
342  evt.window = pChildFrame->wID;
343  evt.message_type = aRESIZE_NOTIFY;
344  evt.format = 32; // always
345  evt.data.l[0] = iL;
346  evt.data.l[1] = iT;
347  evt.data.l[2] = iL + iW; // right
348  evt.data.l[3] = iT + iH; // bottom
349 
350  pChildFrame->pUserCallback(evt.window, (XEvent *)&evt);
351  }
352 }
