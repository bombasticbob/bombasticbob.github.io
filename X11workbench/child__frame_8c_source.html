144  pChildFrames = pChildFrame;

193  return -4; // can't create, SDI

197

200  pChildFrame->destructor = NULL; // make sure
201 
202  if(rFontSet == None)
203  {
204  pChildFrame->rFontSet = None;
205  }
206  else
207  {
208  pChildFrame->rFontSet = WBCopyModifyFontSet(pDisplay, rFontSet, 0, 0);
209  }
210 
211  if(szFocusMenu)
212  {
213  pChildFrame->pszMenuResource = WBCopyString(szFocusMenu);
214 
215  if(!pChildFrame->pszMenuResource)
216  {
217  FWDestroyChildFrame(pChildFrame); // this unhooks everything (alloc'd things are zero'd by bzero)
218  return -5; // not enough memory
219  }
220  }
221  else
222  {
223  pChildFrame->pszMenuResource = NULL;
224  }
225

227 
228  if(pHandlerArray)
229  {
230  const WBFWMenuHandler *pH;
231  int iN;
232 
233  // count the number of entries in the menu handler
234  for(iN=0, pH = pHandlerArray; pH->lMenuID || pH->callback || pH->UIcallback; iN++, pH++)
235  {
236  // NOTHING inside the loop. just count.
237  }
238 
239  // allocate space and make a copy
240 
241  pChildFrame->pMenuHandler = (WBFWMenuHandler *)WBAlloc(sizeof(WBFWMenuHandler)
242  * (iN + 1
243  + sizeof(aChildFrameMenuHandler)
244  / sizeof(aChildFrameMenuHandler[0])
245  )
246  );
247 
248  if(pChildFrame->pMenuHandler)
249  {
250  if(iN > 0)
251  {
252  memcpy(pChildFrame->pMenuHandler, pHandlerArray,
253  sizeof(WBFWMenuHandler) * iN);
254  }
255 
256  memcpy(pChildFrame->pMenuHandler + iN,
257  aChildFrameMenuHandler,
258  sizeof(aChildFrameMenuHandler)); // this includes the NULL entry at the end
259  }
260  else
261  {
262  FWDestroyChildFrame(pChildFrame); // this unhooks everything (alloc'd things are zero'd by bzero)
263  return -5; // not enough memory
264  }
265  }
266 
267  // Next, I need to create the window via 'WBCreateWindow()'
268 
269  bzero(&xswa, sizeof(xswa));
270 
271  xswa.border_pixel = FWGetDefaultBD().pixel;

278  pOwner->iClientWidth - 2, pOwner->iClientHeight - 2, // border is 1 pixel, so subtract 2
279  1, InputOutput,
280  CWBorderPixel | CWBackPixel | CWColormap | CWBitGravity,
281  &xswa);
282 
283  if(pChildFrame->wID == None)
284  {
285  FWDestroyChildFrame(pChildFrame);
286  return -6; // could not create the window
287  }
288 
289  // immediately identify this window's data struct using window data
290  WBSetWindowData(pChildFrame->wID, 0, (void *)pChildFrame); // window data is pointer to this struct
291  pChildFrame->pUserCallback = pUserCallback;
292 
293 
294  // create the default graphics context using the frame's color info
295  WBCreateWindowDefaultGC(pChildFrame->wID, FWGetDefaultFG().pixel, FWGetDefaultBG().pixel);
296 
297  // now allow specific kinds of input messages. I won't need 'structure notify' as they won't work
298  // anyway. those end up going to the frame window, and it passes them (as needed) to me.
299  XSelectInput(pDisplay, pChildFrame->wID,
300  WB_STANDARD_INPUT_MASK | WB_MOUSE_INPUT_MASK | WB_KEYBOARD_INPUT_MASK);
301 
302 
303  // FINALLY, inform the frame window, which will set up some other stuff for me
304 
305  pChildFrame->pOwner = pOwner;
306  iRval = FWAddContainedWindow(pOwner, pChildFrame); // TODO: make THIS assign the owner pointer?
307 
308  if(iRval < 0) // error? (returns negative on error, otherwise the tab index)
309  {
310  FWDestroyChildFrame(pChildFrame); // this frees up all of the resources (but does not free the mem block)
311  }
312 
313  // TODO: other things
314 
315  return iRval;

318 void FWDestroyChildFrame(WBChildFrame *pChildFrame)
319 {
320 WBChildFrame *pC, *pC2;
321 Window wID;
322 
323 
324  if(!pChildFrame)
325  {
326  WB_ERROR_PRINT("ERROR: %s - pChildFrame not valid\n", __FUNCTION__);
327 
328  return;
329  }
330 
331  wID = pChildFrame->wID; // make a copy of it
332 
333  // FIRST, tell the owner to remove me from the list
334 
335  if(pChildFrame->pOwner)
336  {
337  FWRemoveContainedWindow(pChildFrame->pOwner, pChildFrame);
338 
339  if(pChildFrame->pOwner != NULL)
340  {
341  WB_ERROR_PRINT("ERROR: %s - owner pointer was nut 'NULL'd out\n", __FUNCTION__);
342  }
343 
344  pChildFrame->pOwner = NULL;
345  }
346 
347  // NOW, free up any resources that were WBAlloc'd - they must also be free'd
348 
349  if(pChildFrame->szDisplayName)
350  {
351  WBFree(pChildFrame->szDisplayName);
352  pChildFrame->szDisplayName = NULL;
353  }
354 
355  if(pChildFrame->rFontSet)
356  {
357  XFreeFontSet(WBGetDefaultDisplay(), pChildFrame->rFontSet);
358  pChildFrame->rFontSet = None;
359  }
360 
361  if(pChildFrame->pMenuHandler)
362  {
363  WBFree(pChildFrame->pMenuHandler);
364  pChildFrame->pMenuHandler = NULL;
365  }
366 
367  if(pChildFrame->szStatusText)
368  {
369  WBFree(pChildFrame->szStatusText);
370  pChildFrame->szStatusText = NULL;
371  }
372 
373  if(pChildFrame->pszMenuResource)
374  {
375  WBFree(pChildFrame->pszMenuResource);
376  pChildFrame->pszMenuResource = NULL;
377  }
378 
379  // And, NOW, destroy the window itself [if it's valid]
380  //
381  // NOTE: to prevent recursion problems, a 'DestroyNotify' would mark the window ID 'None'
382  // before calling this function. this will happen within my OWN proc, if the superclass doesn't do it
383 
384  if(pChildFrame->wID != None) // if there's a chance of recursion, I detect it here
385  {
386  WBDestroyWindow(pChildFrame->wID);
387  pChildFrame->wID = None; // make sure
388  }
389 
390  // FINALLY, unlink this object from the child frame list. Do it by walking the list, looking for
391  // this entry in the list. If I find it, I unlink it [otherwise, the 'pNext' link might be *stale*]
392 
393  if(pChildFrames) // just in case, walk the list - do not rely on possibly stale pointers
394  {
395  pC2 = NULL;
396  pC = pChildFrames;
397 
398  while(pC)
399  {
400  if(pC == pChildFrame) // just in case
401  {
402  if(pC2)
403  {
404  pC2->pNext = pC->pNext; // remove from link list
405  }
406  else
407  {
408  pChildFrames = pC->pNext; // it was the head of the chain
409  }
410 
411  pC->pNext = NULL;
412 
413  break;
414  }
415 
416  pC2 = pC;
417  pC = pC->pNext;
418  }
419  }
420 
421  pChildFrame->pNext = NULL; // make sure it's not a pointer someplace that can cause problems
422 
423  // if the 'superclass' destructor was specified, call it NOW.
424 
425  if(pChildFrame->destructor)
426  {
427  // at this point, we ALWAYS destroy the callback function.
428  // "it is assumed" that I won't be needing 'DestroyNotify' to
429  // trigger any destruction. That happens *AFTER* the window was
430  // destroyed, and happens asynchronously, typically before the
431  // frame window was destroyed.
432 
433  // so, to make sure that I don't get an event callback that references
434  // the window object after it's free'd, I need to remove the callback.
435 
436  if(wID != None)
437  {
438  WBUnregisterWindowCallback(wID); // no more event handling
439  WBSetWindowData(wID, 0, (void *)NULL); // window data is no longer pointing here
440  }
441 
442  // and THEN, it's safe to call the destructor on the original object.
443  pChildFrame->destructor(pChildFrame); // this should auto-delete as well
444  }
445  else
446  {
447  // TODO: should I perform a 'Destroy Notify' callback of my own?
448 
449  // NOTE: NOT calling 'WBunregisterWindowCallback()' so 'Destroy Notify' does its job. hopefully.
450  }
451 
452  // NOTE: I have to leave the tag alone at this point, to validate that it's
453  // a 'Child Frame' structure. But everything else needs to be zero'd out
454  // (I used to use 'bzero' but I'm just commenting it here, for reference)
455  // so... make sure everything that's free'd gets ZERO'd as well. this is
456  // because a 'DestroyNotify' handler that frees up the memory (and destroys
457  // any 'superclass') will probably call this function again, and it might
458  // happen asynchronously. In fact, it probably WILL happen asynchronously.
459 
460  // bzero(pChildFrame, sizeof(*pChildFrame)); // zero it out (as a matter of course)
461 
462  // note that I do *NOT* free the WBChildFrame HERE - this simply releases all resources for it.
463  // If there is NO DESTRUCTOR, the 'Destroy Notify' handler needs to do that, or at least SOMETHING.
464  // *NOT* handling this properly can result in 'use after free' problems.
465 }
