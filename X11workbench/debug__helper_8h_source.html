274 #if defined(__GNUC__) || defined(_MSVC_VER)
275 static __inline__ int WBCheckDebugLevel(unsigned int dwLevel)
276 {
277 extern unsigned int iWBDebugLevel;
278 
279  if(WB_LIKELY((iWBDebugLevel & DebugLevel_MASK) < (dwLevel & DebugLevel_MASK)))
280  {
281  return 0;
282  }
283 
284  if(!WB_UNLIKELY( WBGetDebugLevel() & DebugSubSystem_RESTRICT )) // RESTRICT not specified
285  {
286  if(!(dwLevel & DebugSubSystem_MASK) ) // no subsystem specified in debug output
287  {
288  return 1; // this is acceptable - since no subsystem specified, allow debug output if not 'RESTRICT'
289  }
290  }
291 
292  // at this point I have a debug subsystem 'RESTRICT' specified
293 
294  if(((dwLevel & DebugSubSystem_MASK) & (iWBDebugLevel & DebugSubSystem_MASK))
295  != 0) // check to see that subsystem bits in 'dwLevel' match bits in 'iWBDebugLevel'
296  {
297  // at least one subsystem bit matches from 'dwLevel' and iWBDebugLevel
298  return 1;
299  }
300 
301  return 0;
302 }
