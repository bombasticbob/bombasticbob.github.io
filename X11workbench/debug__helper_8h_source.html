300 #if defined(__GNUC__) || defined(_MSVC_VER)
301 static __inline__ int WBCheckDebugLevel(WB_UINT64 dwLevel)
302 {
303 extern WB_UINT64 iWBDebugLevel;
304 
305  if(WB_LIKELY((iWBDebugLevel & DebugLevel_MASK) < (dwLevel & DebugLevel_MASK)))
306  {
307  return 0;
308  }
309 
310  if(!WB_UNLIKELY( WBGetDebugLevel() & DebugSubSystem_RESTRICT )) // RESTRICT not specified
311  {
312  if(!(dwLevel & DebugSubSystem_MASK) ) // no subsystem specified in debug output
313  {
314  return 1; // this is acceptable - since no subsystem specified, allow debug output if not 'RESTRICT'
315  }
316  }
317 
318  // at this point I have a debug subsystem 'RESTRICT' specified
319 
320  if(((dwLevel & DebugSubSystem_MASK) & (iWBDebugLevel & DebugSubSystem_MASK))
321  != 0) // check to see that subsystem bits in 'dwLevel' match bits in 'iWBDebugLevel'
322  {
323  // at least one subsystem bit matches from 'dwLevel' and iWBDebugLevel
324  return 1;
325  }
326 
327  return 0;
328 }
329 #else // !defined(__GNUC__) && !defined(_MSVC_VER)
330 int WBCheckDebugLevel(WB_UINT64 dwLevel);
331 #endif // defined(__GNUC__) || defined(_MSVC_VER)
