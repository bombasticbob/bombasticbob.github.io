239 static __inline__ int WBCheckDebugLevel(unsigned int dwLevel)
240 {
241 extern unsigned int iWBDebugLevel;
242 
243  if(WB_LIKELY((iWBDebugLevel & DebugLevel_MASK) < (dwLevel & DebugLevel_MASK)))
244  {
245  return 0;
246  }
247 
248  if(!WB_UNLIKELY( WBGetDebugLevel() & DebugSubSystem_RESTRICT )) // RESTRICT not specified
249  {
250  if(!(dwLevel & DebugSubSystem_MASK) ) // no subsystem specified in debug output
251  {
252  return 1; // this is acceptable - since no subsystem specified, allow debug output if not 'RESTRICT'
253  }
254  }
255 
256  // at this point I have a debug subsystem 'RESTRICT' specified
257 
258  if(((dwLevel & DebugSubSystem_MASK) & (iWBDebugLevel & DebugSubSystem_MASK))
259  != 0) // check to see that subsystem bits in 'dwLevel' match bits in 'iWBDebugLevel'
260  {
261  // at least one subsystem bit matches from 'dwLevel' and iWBDebugLevel
262  return 1;
263  }
264 
265  return 0;
266 }
