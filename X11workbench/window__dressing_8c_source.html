321 void WBCalcVScrollBar(WB_SCROLLINFO *pScrollInfo, WB_GEOM *pgeomClient, int iVScrollWidth,
322  int iHScrollHeight, int nListItems, int nPos)
323 {
324 int iKnobSize, iKnobPos, iBarHeight, iBarWidth;
325 int i1, i2;
326 
327  iBarHeight = pgeomClient->height;
328  iBarWidth = pgeomClient->width;
329 
330  if(pScrollInfo->iVBarHeight == iBarHeight &&
331  pScrollInfo->iHBarWidth == iBarWidth &&
332  pScrollInfo->iVBarHeight == iBarHeight &&
333  pScrollInfo->iHBarWidth == iBarWidth &&
334  pScrollInfo->iVKnob >= 0 &&
335  pScrollInfo->iVKnobSize >= 0 &&
336  pScrollInfo->iVPos == nPos &&
337  pScrollInfo->iVMin == 0 &&
338  pScrollInfo->iVMax == (nListItems - 1))
339  {
340  // assume it's set up correctly
341 
342 // WB_ERROR_TEXT("TEMPORARY: short cycling calc scroll info\n");
343 // return;
344  }
345 
346  if(pScrollInfo->iVBarHeight != iBarHeight ||
347  pScrollInfo->iHBarWidth != iBarWidth ||
348  pScrollInfo->iVBarHeight != iBarHeight ||
349  pScrollInfo->iHBarWidth != iBarWidth)
350  {
351  // client size change detected
352  // zero out the H Scroll info so that it forces re-calculation next time
353  pScrollInfo->iHKnob = pScrollInfo->iHKnobSize = pScrollInfo->iHPos = pScrollInfo->iHMin = pScrollInfo->iHMax = 0;
354  }
355 
356 
357 // WB_ERROR_PRINT("TEMPORARY: %d %d %d %d %d\n",
358 // iBarHeight, iVScrollWidth, iHScrollHeight, nListItems, nPos);
359 
360  pScrollInfo->iVScrollWidth = iVScrollWidth;
361  pScrollInfo->iHScrollHeight = iHScrollHeight;
362  pScrollInfo->iVBarHeight = iBarHeight;
363  pScrollInfo->iHBarWidth = iBarWidth;
364 
365  pScrollInfo->iVMin = 0;
366  pScrollInfo->iVMax = nListItems - 1;
367  if(nPos == -1 ||
368  (nPos >= 0 && nPos < nListItems))
369  {
370  pScrollInfo->iVPos = nPos;
371  }
372  else
373  {
374  pScrollInfo->iVPos = nPos = -1;
375  }
376 
377  // knob height equals 'nListItems / nListItems!' multiplied by
378  // the available knob height, for a minimum value of 'iHScrollHeight'
379 
380  iKnobSize = (iBarHeight - 4 * iHScrollHeight - 2);
381  if(iKnobSize > iHScrollHeight / 2)
382  {
383  i1 = iHScrollHeight * (2 * iHScrollHeight + 1); // 'twice scroll height' factorial
384  i2 = iHScrollHeight * 2;
385 
386  while(i2 > 2 && i1 > iKnobSize) // using the above starting point get 'max factorial less than height'
387  {
388  i1 -= i2;
389  i2--;
390  }
391 
392  if(nListItems >= i2) // more than twice scroll height?
393  {
394  if(nListItems > iKnobSize || iKnobSize - i1 < iHScrollHeight / 2)
395  {
396  iKnobSize = -1; // to fix later
397  }
398  else
399  {
400  iKnobSize -= i1; // twice scroll height factorial
401  iKnobSize -= (nListItems - iHScrollHeight * 2 + 1);
402  }
403  }
404  else
405  {
406  iKnobSize -= i1 // 'twice knob height' factorial
407  - (nListItems + 1) * nListItems / 2; // 'nListItems' factorial
408  }
409 
410  if(iKnobSize < 3 * iHScrollHeight / 4)
411  {
412  iKnobSize = 3 * iHScrollHeight / 4; // minimum size
413  }
414  }
415  else
416  {
417  iKnobSize = 3 * iHScrollHeight / 4; // worst case use this anyway
418  }
419 
420  // cache these geometries because I need them for mouse handling
421 
422  pScrollInfo->geomVBar.y = pgeomClient->y + 1;
423  pScrollInfo->geomVBar.height = pgeomClient->height - 2;
424  pScrollInfo->geomVBar.width = pScrollInfo->iVScrollWidth;
425  pScrollInfo->geomVBar.x = pgeomClient->x + pgeomClient->width
426  - pScrollInfo->geomVBar.width - 1;
427 
428  pScrollInfo->geomVDown.x = pScrollInfo->geomVUp.x
429  = pScrollInfo->geomVBar.x + 1;
430  pScrollInfo->geomVDown.width = pScrollInfo->geomVUp.width
431  = pScrollInfo->geomVBar.width - 2;
432  pScrollInfo->geomVDown.height = pScrollInfo->geomVUp.height
433  = pScrollInfo->iHScrollHeight - 1;
434 
435  pScrollInfo->geomVUp.y = pScrollInfo->geomVBar.y + 1;
436  pScrollInfo->geomVDown.y = pScrollInfo->geomVBar.y
437  + pScrollInfo->geomVBar.height
438  - pScrollInfo->geomVDown.height - 1;
439 
440  pScrollInfo->geomVKnob.x = pScrollInfo->geomVDown.x;
441  pScrollInfo->geomVKnob.width = pScrollInfo->geomVDown.width;
442  pScrollInfo->geomVKnob.height = pScrollInfo->iVKnobSize
443  = iKnobSize;
444 
445  // now that I have a reasonable estimate of the knob size, calculate
446  // the starting vertical position of the knob
447 
448  if(nPos <= 0 || nListItems <= 0)
449  {
450  iKnobPos = 0;
451  }
452  else
453  {
454  int iBarScrollArea = pScrollInfo->geomVDown.y
455  - (pScrollInfo->geomVUp.y + pScrollInfo->geomVUp.height)
456  - iKnobSize
457  - 1; // inclusive value
458 
459  if(nPos >= (nListItems - 1))
460  {
461  iKnobPos = iBarScrollArea;
462  }
463  else
464  {
465  iKnobPos = (WB_INT64)nPos * (WB_INT64)iBarScrollArea
466  / (WB_INT64)(nListItems - 1);
467  }
468  }
469 
470 
471  pScrollInfo->geomVKnob.y = (pScrollInfo->iVKnob = iKnobPos)
472  + pScrollInfo->geomVUp.y + pScrollInfo->geomVUp.height;
473 
474  // fixed values go here
475  pScrollInfo->geomVBar.border = 1;
476  pScrollInfo->geomVUp.border = 0;
477  pScrollInfo->geomVDown.border = 0;
478  pScrollInfo->geomVKnob.border = 0;
479 
480 }
