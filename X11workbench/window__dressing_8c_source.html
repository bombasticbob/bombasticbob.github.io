274 static void InternalCalcVScrollBar(WB_SCROLLINFO *pScrollInfo, WB_GEOM *pgeomClient, int iVScrollWidth, int iHScrollHeight)
275 {
276 int iKnobSize, iKnobPos, iBarHeight;
277 int i1, i2;
278 int nListItems;
279 
280 
281  iBarHeight = pgeomClient->height;
282 
283  pScrollInfo->iVScrollWidth = iVScrollWidth;
284  pScrollInfo->iVBarHeight = iBarHeight;
285 
286  WB_DEBUG_PRINT(DebugLevel_Medium | DebugSubSystem_ScrollBar | DebugSubSystem_Expose,
287  "%s - iVScrollWidth=%d, iHScrollHeight=%d iBarHeight=%d\n",
288  __FUNCTION__, iVScrollWidth, iHScrollHeight, iBarHeight);
289 
290  if(pScrollInfo->iVPos < pScrollInfo->iVMin || pScrollInfo->iVPos > pScrollInfo->iVMax)
291  {
292  pScrollInfo->iVPos = -1; // for now
293  }
294 
295  nListItems = pScrollInfo->iVMax - pScrollInfo->iVMin + 1;
296 
297 
298  // knob height equals 'nListItems / nListItems!' multiplied by
299  // the available knob height, for a minimum value of 'iHScrollHeight'
300 
301  iKnobSize = (iBarHeight - 4 * iHScrollHeight - 2);
302 
303  if(iKnobSize > iHScrollHeight / 2)
304  {
305  i1 = iHScrollHeight * (2 * iHScrollHeight + 1); // 'twice scroll height' factorial
306  i2 = iHScrollHeight * 2;
307 
308  while(i2 > 2 && i1 > iKnobSize) // using the above starting point get 'max factorial less than height'
309  {
310  i1 -= i2;
311  i2--;
312  }
313 
314  if(nListItems >= i2) // more than twice scroll height?
315  {
316  if(nListItems > iKnobSize || iKnobSize - i1 < iHScrollHeight / 2)
317  {
318  iKnobSize = -1; // to fix later
319  }
320  else
321  {
322  iKnobSize -= i1; // twice scroll height factorial
323  iKnobSize -= (nListItems - iHScrollHeight * 2 + 1);
324  }
325  }
326  else
327  {
328  iKnobSize -= i1 // 'twice knob height' factorial
329  - (nListItems + 1) * nListItems / 2; // 'nListItems' factorial
330  }
331 
332  if(iKnobSize < 3 * iHScrollHeight / 4)
333  {
334  iKnobSize = 3 * iHScrollHeight / 4; // minimum size
335  }
336  }
337  else
338  {
339  iKnobSize = 3 * iHScrollHeight / 4; // worst case use this anyway
340  }
341 
342  // cache these geometries because I need them for mouse handling
343 
344  pScrollInfo->iVKnobSize = iKnobSize;
345 
346  pScrollInfo->geomVBar.x = pgeomClient->x
347  + pgeomClient->width
348  - iVScrollWidth
349  + 2 * pgeomClient->border; // to account for the border size
350 
351  pScrollInfo->geomVBar.y = pgeomClient->y
352  - pgeomClient->border; // to account for the border size
353  pScrollInfo->geomVBar.height = pgeomClient->height
354  + 2 * pgeomClient->border; // to account for the border size
355  pScrollInfo->geomVBar.width = iVScrollWidth;
356 
357  // is there a vertical scroll bar? If so, I need to adjust the width
358  if(pScrollInfo->iHMin < pScrollInfo->iHMax) // not '-1'
359  {
360  pScrollInfo->geomVBar.height -= iHScrollHeight; // make it a tad smaller
361  }
362 
363 
364  pScrollInfo->geomVUp.x = pScrollInfo->geomVBar.x + 1;
365  pScrollInfo->geomVUp.y = pScrollInfo->geomVBar.y + 1;
366  pScrollInfo->geomVUp.width = pScrollInfo->geomVBar.width - 2;
367  pScrollInfo->geomVUp.height = pScrollInfo->geomVUp.width;
368 
369  pScrollInfo->geomVDown.x = pScrollInfo->geomVUp.x;
370  pScrollInfo->geomVDown.y = pScrollInfo->geomVBar.y
371  + pScrollInfo->geomVBar.height
372  - pScrollInfo->geomVDown.height
373  - 1;
374  pScrollInfo->geomVDown.width = pScrollInfo->geomVUp.width;
375  pScrollInfo->geomVDown.height = pScrollInfo->geomVUp.height;
376 
377  pScrollInfo->geomVKnob.x = pScrollInfo->geomVDown.x;
378  pScrollInfo->geomVKnob.width = pScrollInfo->geomVDown.width;
379  pScrollInfo->geomVKnob.height = iKnobSize;
380 
381  // now that I have a reasonable estimate of the knob size, calculate
382  // the starting vertical position of the knob
383 
384  if(pScrollInfo->iVPos <= 0 || nListItems <= 0)
385  {
386  iKnobPos = 0;
387  }
388  else
389  {
390  int iBarScrollArea = pScrollInfo->geomVDown.y
391  - (pScrollInfo->geomVUp.y + pScrollInfo->geomVUp.height)
392  - iKnobSize
393  - 1; // inclusive value
394 
395  if(pScrollInfo->iVPos >= (nListItems - 1))
396  {
397  iKnobPos = iBarScrollArea;
398  }
399  else
400  {
401  iKnobPos = (WB_INT64)pScrollInfo->iVPos * (WB_INT64)iBarScrollArea
402  / (WB_INT64)(nListItems - 1);
403  }
404  }
405 
406 
407  pScrollInfo->geomVKnob.y = (pScrollInfo->iVKnob = iKnobPos)
408  + pScrollInfo->geomVUp.y + pScrollInfo->geomVUp.height;
409 
410  // fixed values go here
411  pScrollInfo->geomVBar.border = 1;
412  pScrollInfo->geomVUp.border = 0;
413  pScrollInfo->geomVDown.border = 0;
414  pScrollInfo->geomVKnob.border = 0;
415 
416 }
