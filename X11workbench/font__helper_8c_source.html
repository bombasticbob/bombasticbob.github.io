111 WB_FONT WBCopyFont(Display *pDisplay, WB_FONTC pOldFont)
112 {
113 WB_FONT pRval;
114 
115  if(!pDisplay)
116  {
117  pDisplay = WBGetDefaultDisplay();
118  }
119 
120  if(!pOldFont)
121  {
122  pOldFont = WBGetDefaultFont();
123 
124  if(!pOldFont)
125  {
126  WB_WARN_PRINT("%s returns NULL (WBGetDefaultFont returns NULL)\n", __FUNCTION__);
127 
128  return NULL;
129  }
130  }
131 
132  pRval = (WB_FONT)WBAlloc(sizeof(*pRval));
133  if(pRval)
134  {
135  memset(pRval, 0, sizeof(*pRval));
136 
137  pRval->pDisplay = pDisplay;
138  pRval->iAscent = pRval->iDescent = pRval->iHeight = pRval->iAvgCharWidth = pRval->iMaxCharWidth = -1;
139 
140 #ifdef X11WORKBENCH_TOOLKIT_HAVE_XFT
141  if(pOldFont->pxftFont)
142  {
143  pRval->pxftFont = XftFontCopy(pDisplay, pOldFont->pxftFont);
144  if(!pRval->pxftFont)
145  {
146  goto bad_font;
147  }
148 // pRval->pxftFontInfo = copy something from old one
149  }
150  else
151 #endif // X11WORKBENCH_TOOLKIT_HAVE_XFT
152  {
153  if(pOldFont->fsFont != None)
154  {
155  pRval->fsFont = WBCopyModifyFontSet(pDisplay, pOldFont->fsFont, 0, 0);
156 
157  if(pRval->fsFont == None)
158  {
159  WB_WARN_PRINT("%s returns NULL (unable to copy font set)\n", __FUNCTION__);
160 
161  goto bad_font;
162  }
163  }
164  /*else*/ if(pOldFont->pFontStruct)
165  {
166  pRval->pFontStruct = WBCopyFontX(pOldFont->pFontStruct);
167  if(!pRval->pFontStruct)
168  {
169  WB_WARN_PRINT("%s returns NULL (unable to copy font struct)\n", __FUNCTION__);
170 
171  goto bad_font;
172  }
173  }
174 
175  if(!pOldFont->pFontStruct && pOldFont->fsFont == None)
176  {
177  WB_WARN_PRINT("%s returns NULL (bad font)\n", __FUNCTION__);
178 bad_font:
179  WBFreeFont(pDisplay, pRval);
180 
181  return NULL;
182  }
183  }
184  }
185  else
186  {
187  WB_WARN_PRINT("%s returns NULL (no memory for structure)\n", __FUNCTION__);
188  }
189 
190  return pRval;
191 }
