279 static int InternalFontMatch(const WB_FONT_INFO *p1, const WB_FONT_INFO *p2)
280 {
281  int iRval = 0;
282 
283  if(!strcasecmp(p1->szFoundry, p2->szFoundry) ||
284  p1->szFoundry[0] == '*' || p2->szFoundry[0] == '*' ||
285  !p1->szFoundry[0] || !p2->szFoundry[0])
286  {
287  iRval += 3; // score 3 if the foundry matches
288  }
289 
290  if(!strcasecmp(p1->szFamily, p2->szFamily) ||
291  p1->szFamily[0] == '*' || p2->szFamily[0] == '*')
292  {
293  iRval += 10; // score 10 for font name match
294  }
295 
296  if(p1->iWeight == -1 || p2->iWeight == -1 || p1->iWeight == p2->iWeight)
297  iRval += 5; // score 5 for matching weight
298  else
299  iRval += 3 - abs(p1->iWeight - p2->iWeight); // sort of fuzzy matching
300 
301  if(p1->iSlant == -1 || p2->iSlant == -1 || p1->iSlant == p2->iSlant)
302  iRval += 5; // score 5 for matching Slant
303  else
304  iRval += 3 - abs(p1->iSlant - p2->iSlant); // sort of fuzzy matching
305 
306  if(p1->iWidth == -1 || p2->iWidth == -1 || p1->iWidth == p2->iWidth)
307  iRval += 2; // score 2 for matching Width
308 
309  // for 'szAdStyle' match, it's 'sans' 'ja' 'ko' or blank
310  // since blank may match any of them (or none) it's a 2 point delta for 'sans', 1 otherwise
311 
312  if(!strcasecmp(p1->szAdStyle,p2->szAdStyle) ||
313  p1->szAdStyle[0] == '*' || p2->szAdStyle[0] == '*')
314  {
315  iRval += 4; // score 4 for matching style (sans, ko, ja, blank)
316  }
317  else if((!p1->szAdStyle[0] || !p1->szAdStyle[1]) &&
318  (!strcasecmp(p1->szAdStyle,"sans") || !strcasecmp(p2->szAdStyle,"sans")))
319  {
320  iRval += 2; // score 2 for 'sort of matching' (blank vs sans)
321  }
322  else if(!p1->szAdStyle[0] || !p1->szAdStyle[1])
323  {
324  iRval ++; // score 1 for 'sort of matching' (blank vs something other than sans)
325  }
326  else
327  {
328  iRval -= 2; // subtract 2 if there's any other combination
329  }
330 
331  if(p1->iPixelSize == -1 || p2->iPixelSize == -1 || p1->iPixelSize == p2->iPixelSize)
332  iRval += 3; // score 3 for matching PixelSize
333  else
334  iRval += 2 - abs(p1->iPixelSize - p2->iPixelSize); // sort of fuzzy matching
335 
336  if(p1->iPointSize == -1 || p2->iPointSize == -1 || p1->iPointSize == p2->iPointSize)
337  iRval += 3; // score 3 for matching PointSize
338  else
339  iRval += 2 - abs(p1->iPointSize - p2->iPointSize) / 10; // sort of fuzzy matching
340 
341  if((p1->iResX == p2->iResX && p1->iResY == p2->iResY) ||
342  (p1->iResX == -1 && p1->iResY == -1) ||
343  (p2->iResX == -1 && p2->iResY == -1))
344  {
345  iRval++; // 1 point for resolution (exact match only)
346  }
347 
348  if(p1->iSpacing == -1 || p2->iSpacing == -1 || p1->iSpacing == p2->iSpacing)
349  iRval += 5; // 5 points if spacing matches
350  else if(p1->iSpacing == 0 || p2->iSpacing == 0)
351  iRval += 3; // 3 points if either is unspecified but not wildcard
352  else
353  iRval += 3 - abs(p1->iSpacing - p2->iSpacing) / 5; // 1/5 of the delta (somewhat relative to point sizes)
354 
355  if(!strcasecmp(p1->szRegistry,p2->szRegistry) ||
356  !p1->szRegistry[0] || !p2->szRegistry[0] ||
357  p1->szRegistry[0] == '*' || p2->szRegistry[0] == '*')
358  {
359  iRval += 4; // score 4 for matching registry (iso8859 etc.)
360  }
361 
362  if(!strcasecmp(p1->szEncoding,p2->szEncoding) ||
363  !p1->szEncoding[0] || !p2->szEncoding[0] ||
364  p1->szEncoding[0] == '*' || p2->szEncoding[0] == '*')
365  {
366  iRval += 4; // score 4 for matching character encoding
367  }
368 
369  return iRval;
370 }

836 XFontStruct *WBLoadFont(Display *pDisplay, const char *szFontName,
837  int iFontSize, int iFlags)
838 {
839 XFontStruct *pRval=NULL, *pFSInfo=NULL;
840 char **ppNames;
841 int i1, i2, i3, iBest, iCount = 0, iWildcardFlag = 0;
842 char tbuf[512], tbuf2[512], tbuf3[512];
843 
844 // font names follow one of these patterns...
845 //
846 // AliasName (not preceded by a '-')
847 //
848 // or
849 //
850 // -foundry-family-weight-slant-sWidth-adstyle-pixelsize-pointsize-resX-resY-spacing-avgwidth-registry-encoding
851 //
852 // foundry = adoby, b&h, xfree86, etc.
853 // family = lucida, terminal, courier, etc.
854 // weight = bold, demibold, medium, regular
855 // slant = i, o, r (italic, oblique???, regular???)
856 // sWidth = normal, semicondensed
857 // adstyle = (blank), sans, ja, ko, ???
858 // pixelsize = height in pixels
859 // pointsize = width in 10*points
860 // resX, resY = resolution?
861 // spacing = c, m, p (condensed, medium, proportional)
862 // avgWidth (in points?)
863 // registry (i.e. 'iso8859' etc.)
864 // encoding (the '-' value following the registry, such as '1' for 'iso8859-1')
865 
866 
867  // step 1: if it's a font alias, get the REAL font info. I'll need to do this if I'm requesting
868  // a bold or italic version of an aliased font, or even a different size.
869 
870  if(szFontName && *szFontName && !strchr(szFontName, '*') && !strchr(szFontName,'-'))
871  {
872  // get the alias name
873  WBFontNameFromAlias(szFontName, tbuf, sizeof(tbuf));
874 
875  if(!*tbuf)
876  {
877  strcpy(tbuf, "-*-");
878  strncpy(tbuf + 3, szFontName, sizeof(tbuf) - 3);
879  }
880  }
881  else
882  {
883  iWildcardFlag = 1; // additional information
884 
885  strncpy(tbuf, szFontName, sizeof(tbuf));
886  }
887 
888  // at this point 'tbuf' is assumed to be a properly formatted font string
889 
890  if(iFlags || iFontSize)
891  {
892  InternalBuildFontString(tbuf, iFontSize, iFlags, tbuf2, sizeof(tbuf2), FALSE);
893 // fprintf(stderr, "TEMPORARY: font string %s\n", tbuf2);
894  }
895  else
896  strncpy(tbuf2, tbuf, sizeof(tbuf2));
897 
898  ppNames = XListFontsWithInfo(pDisplay, tbuf2, 2, &iCount, &pFSInfo);
899 
900  // in this case if I only have one font returned, and there are no overrides in iFlags,
901  // then that's the one I use. This is the simplest form of font mapping
902 
903  if(iCount == 1 && ppNames) // there is only a single match
904  pRval = XLoadQueryFont(pDisplay, *ppNames);
905 
906  if(ppNames)
907  {
908  XFreeFontInfo(ppNames, pFSInfo, iCount);
909  ppNames = NULL;
910  pFSInfo = NULL;
911  iCount = 0;
912  }
913 
914  if(pRval)
915  return pRval;
916 
917  // no match or multi-match - use wildcards, pick "the best one"
918 
919  InternalBuildFontString(tbuf, iFontSize, iFlags, tbuf3, sizeof(tbuf3), TRUE);
920 // fprintf(stderr, "TEMPORARY: font string 2 %s\n", tbuf3);
921 
922  ppNames = XListFontsWithInfo(pDisplay, tbuf3, 8192, &iCount, &pFSInfo);
923 
924  if(ppNames && iCount == 1)
925  {
926  pRval = XLoadQueryFont(pDisplay, *ppNames);
927  }
928  else if(ppNames && iCount > 1)
929  {
930  iBest = -1;
931  for(i1 = 0, i2 = -1; i1 < iCount; i1++)
932  {
933  i3 = InternalFontMatch2(ppNames[i1], tbuf2); // compare with non-forced-wildcard version
934  if(i3 > iBest)
935  {
936  iBest = i3;
937  i2 = i1;
938  }
939  }
940 
941  if(i2 >= 0)
942  {
943  pRval = XLoadQueryFont(pDisplay, ppNames[i2]);
944 // fprintf(stderr, "TEMPORARY 3: best match %s\n", ppNames[i2]);
945  }
946  }
947 
948  if(ppNames)
949  {
950  XFreeFontInfo(ppNames, pFSInfo, iCount);
951  ppNames = NULL;
952  pFSInfo = NULL;
953  iCount = 0;
954  }
955 
956  // TODO: other forms of 'fuzzy' matching?
957 
958  return pRval;
959 }
