248 void MyQuickSort2(struct my_qsort_data *pData, int iLow, int iHigh )
249 {
250 int iUp, iDown, iBreak;
251 void *pBreak;
252 unsigned char buf[32], buf2[32]; // if smaller than 32 bytes, use THIS for temp object storage
253 
254 
255  if( iLow < iHigh ) // do not sort 1 element block
256  {
257  iBreak = (iLow + iHigh) / 2;
258  pBreak = ElementFromIndex(pData, iBreak); // initial pivot point
259 
260  if( (iHigh - iLow) > 5 ) // 5 or more elements?
261  {
262  // do a 'median of 3' optimization when practical
263  // this ensures a better pivot point, limiting the
264  // effect of an already sorted or nearly sorted
265  // array on performance. Pivot points should be
266  // as close to the median value as practical for
267  // the current range of data points. Otherwise,
268  // a series of bad pivot points could degenerate
269  // the 'quicksort' into a 'slowsort', and possibly
270  // cause stack overflows on large data sets.
271 
272  if(DoCompareI(pData, iLow, iHigh) <= 0)
273  {
274  if(DoCompareP(pData, iLow, pBreak) < 0)
275  {
276  if(DoCompareP(pData, iHigh, pBreak) < 0)
277  {
278  pBreak = ElementFromIndex(pData, iHigh);
279  }
280  }
281  else
282  {
283  pBreak = ElementFromIndex(pData, iLow);
284  }
285  }
286  else
287  {
288  if(DoCompareP(pData, iHigh, pBreak) < 0)
289  {
290  if(DoCompareP(pData, iLow, pBreak) < 0)
291  {
292  pBreak = ElementFromIndex(pData, iLow);
293  }
294  }
295  else
296  {
297  pBreak = ElementFromIndex(pData, iHigh);
298  }
299  }
300  }
301 
302  memcpy(pData->pBR, pBreak, pData->cbData); // make copy of it before swapping anything
303  pBreak = pData->pBR; // and now point to the buffer (faster this way)
304 
305  iUp = iLow; // initialize indices
306  iDown = iHigh;
307 
308  do
309  {
310  // Move in from both sides towards the pivot point.
311 
312  // note: it may be possible to thread THESE two loops
313  // though it's likely not to help a whole lot
314 
315  while( iUp < iHigh)
316  {
317  if(DoCompareP(pData, iUp, pBreak) < 0)
318  {
319  iUp++;
320  }
321  else
322  {
323  break;
324  }
325  }
326 
327  while(iDown > iLow)
328  {
329  if(DoCompareP(pData, iDown, pBreak) > 0)
330  {
331  iDown--;
332  }
333  else
334  {
335  break;
336  }
337  }
338 
339 
340  // if low/high boundaries have not crossed, swap current
341  // 'boundary' values so that the 'iUp' pointer points
342  // to a value less than or equal to the pivot point,
343  // and the 'iDown' value points to a value greater than
344  // or equal to the pivot point, and continue.
345 
346  if( iUp <= iDown )
347  {
348  if(iUp != iDown)
349  {
350  DoSwapData(pData, iUp, iDown);
351  }
352 
353  iUp++;
354  iDown--;
355  }
356 
357  } while ( iUp <= iDown );
358 
359  // the recursive part... [NOTE if I ever want to thread this, I can make work units here]
360 
361  if(iLow < iDown ) // everything to the left of the pivot point
362  {
363  MyQuickSort2(pData, iLow, iDown );
364  }
365  if(iUp < iHigh) // everything to the right of the pivot point
366  {
367  MyQuickSort2(pData, iUp, iHigh );
368  }
369  }
370 }

1824  WB_INVALID_FILE_HANDLE, szAppName, va);

1829

1831  close(hP[1]);

1834  }
1835

1890  }
1891  }
