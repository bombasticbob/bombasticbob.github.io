470 static void MBMenuHandleMenuItem(Display *pDisplay, WBMenuBarWindow *pSelf, WBMenu *pMenu, WBMenuItem *pItem)
471 {
472 Window wID = pSelf->wSelf;
473 
474  if(pItem->iAction & WBMENU_POPUP_HIGH_BIT)
475  {
476  XClientMessageEvent evt;
477 
478  // post a high-priority message to myself to display the menu
479 
480  bzero(&evt, sizeof(evt));
481  evt.type = ClientMessage;
482  evt.display = pDisplay;
483  evt.window = wID;
484  evt.message_type = aMENU_DISPLAY_POPUP;
485  evt.format = 32;
486  evt.data.l[0] = pItem->iAction & WBMENU_POPUP_MASK;
487  evt.data.l[1] = pItem->iPosition; // minimum extent
488  evt.data.l[2] = pItem->iPosition + pItem->iTextWidth; // maximum extent
489 
490  WBPostPriorityEvent(wID, (XEvent *)&evt);
491 
492  WB_DEBUG_PRINT(DebugLevel_Chatty | DebugSubSystem_Menu,
493  "%s - posting client event message to display popup menu\n", __FUNCTION__);
494  }
495  else
496  {
497  XClientMessageEvent evt;
498 
499  bzero(&evt, sizeof(evt));
500  evt.type = ClientMessage;
501 
502  evt.display = pDisplay;
503  evt.window = pSelf->wOwner;
504  evt.message_type = aMENU_COMMAND;
505  evt.format = 32; // always
506  evt.data.l[0] = pItem->iAction; // menu command message ID
507  evt.data.l[1] = (long)pMenu; // pointer to menu object
508  evt.data.l[2] = wID; // window ID of menu bar
509 
510  WBPostEvent(pSelf->wOwner, (XEvent *)&evt);
511 
512  pSelf->iSelected = -1; // after handling the message I de-select the menu item
513 
514  WB_DEBUG_PRINT(DebugLevel_WARN | DebugSubSystem_Menu | DebugSubSystem_Event,
515  "%s - Post Event: %08xH %08xH %pH %08xH\n", __FUNCTION__,
516  (int)aMENU_COMMAND, (int)pItem->iAction,
517  pMenu, (int)wID);
518 
520  }
521 }
