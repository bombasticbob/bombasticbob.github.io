800 int WBDialogControlSetDialogProp(WBDialogControl *pCtrl, WB_DIALOG_PROP *pPropVal)
801 {
802 int i1;
803 WBDialogPropList *pPropList;
804 WB_DIALOG_PROP *pProp = NULL;
805 Atom aProp;
806 
807  if(!pCtrl || !pPropVal)
808  {
809  WB_ERROR_PRINT("%s - pCtrl %p or pPropVal %p\n", __FUNCTION__,
810  pCtrl, pPropVal);
811 
812  return -1;
813  }
814 
815  if(!pCtrl->pPropList)
816  {
817  if(!(pCtrl->pPropList = __ConstructPropList()))
818  {
819  WB_ERROR_PRINT("%s - unable to allocate property list for window %d (%08xH)\n",
820  __FUNCTION__, (int)pCtrl->wID, (int)pCtrl->wID);
821 
822  return -2;
823  }
824  }
825 
826  pPropList = pCtrl->pPropList;
827  aProp = pPropVal->aProp;
828 
829  for(i1=0; i1 < pPropList->nProps; i1++)
830  {
831  register WB_DIALOG_PROP *pProp0 = &(pPropList->aDlgProp[i1]);
832 
833  if(pProp0 && pProp0->aProp == aProp)
834  {
835  pProp = pProp0;
836  break;
837  }
838  }
839 
840  if(!pProp)
841  {
842  if((pPropList->nProps + 1) >= pPropList->nMaxProps)
843  {
844  int iNewSize = (pPropList->nMaxProps + INITIAL_PROPERTY_MAX / 2)
845  * sizeof(pPropList->aDlgProp[0])
846  + sizeof(*pPropList);
847 
848  pPropList = (WBDialogPropList *)realloc(pPropList, iNewSize);
849  if(!pPropList)
850  {
851  WB_ERROR_PRINT("%s - unable to RE-allocate property list for window %d (%08xH)\n",
852  __FUNCTION__, (int)pCtrl->wID, (int)pCtrl->wID);
853 
854  return -3;
855  }
856 
857  pCtrl->pPropList = pPropList;
858  pPropList->nMaxProps += INITIAL_PROPERTY_MAX / 2;
859 
860  if(pPropList->nProps >= pPropList->nMaxProps)
861  {
862  WB_ERROR_PRINT("%s - internal inconsistency, nProps=%d, nMaxProps=%d\n",
863  __FUNCTION__, pPropList->nProps, pPropList->nMaxProps);
864  return -4;
865  }
866  }
867 
868  pProp = &(pPropList->aDlgProp[pPropList->nProps]);
869  pPropList->nProps++;
870 
871  pProp->aProp = aProp;
872  pProp->lVal = 0;
873  pProp->pVal = NULL;
874  }
875 
876  pProp->lVal = pPropVal->lVal; // just copy the lVal
877 
878  if(pProp->pVal != pPropVal->pVal) // if it changes... and ONLY if it changes
879  {
880  if(pProp->pVal)
881  {
882  if(pProp->aProp == aDLGC_LISTINFO)
883  {
884  DLGCListInfoDestructor((LISTINFO *)(pProp->pVal));
885  }
886  else
887  {
888  free(pProp->pVal);
889  }
890  }
891 
892  pProp->pVal = pPropVal->pVal;
893  }
894 
895  // last but not least, tell the dialog control its properties were altered
896  // by directly calling its callback function with the 'property change' event
897 
898  WB_DEBUG_PRINT(DebugLevel_Chatty | DebugSubSystem_DialogCtrl | DebugSubSystem_Dialog,
899  "%s - DLGC_PROP_NOTIFY (set) for window %d (%08xH)\n",
900  __FUNCTION__, (int)pCtrl->wID, (int)pCtrl->wID);
901 
902  DLGNotifySelf(pCtrl, aDLGC_PROP_NOTIFY, pPropVal->aProp, 0, 0, 0, 0);
903 
904  return 0; // success
905 }

909 int i1, bNotify = 0;
910 WBDialogPropList *pPropList = pCtrl ? pCtrl->pPropList : NULL;
