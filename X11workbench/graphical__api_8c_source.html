568 
569 
570  pRval = (WBGC)WBAlloc(sizeof(*pRval)); // WBGC is a pointer, don't use 'sizeof' on that!!!
571  if(!pRval)
572  {
573  return NULL;
574  }
575 
576  if(!pDisplay)
577  {
578  pDisplay = WBGetDefaultDisplay();
579  }
580 
581  if(dw == None)
582  {
583  dw = WBGetHiddenHelperWindow(); // basically I need to have something valid for a Drawable
584  }
585 
586  memset(pRval, 0, sizeof(*pRval));
587

590  {
591  memcpy(&(pRval->values), values, sizeof(pRval->values));

593

596

599  pRval->gc = XCreateGC(pDisplay, dw, valuemask, valuemask && values ? &(pRval->values) : NULL);
600  END_XCALL_DEBUG_WRAPPER
601 
602  if(pRval->gc == None)
603  {
604  WBFree(pRval);
605  return NULL;
606  }
607

609  memset(&(pRval->values), 0, sizeof(pRval->values)); // must do this first
610 
611  BEGIN_XCALL_DEBUG_WRAPPER
612  XGetGCValues(pDisplay, pRval->gc,
613  (GCAll & ~(GCClipMask | GCDashList)), // clip mask and dash mask can't be requested

615  END_XCALL_DEBUG_WRAPPER

641 
642  pRval->values.clip_mask = None;
643  }
644

647
